CREATE TABLE dept (
    dept_id INT NOT NULL,
    parent_dept_id INT NULL,
    dept_code VARCHAR(20) NOT NULL,
    dept_name VARCHAR(100) NOT NULL,
    sort_no INT NOT NULL,
    PRIMARY KEY (dept_id),
    UNIQUE KEY uk_dept_code (dept_code),
    KEY idx_dept_parent (parent_dept_id),
    CONSTRAINT fk_dept_parent
        FOREIGN KEY (parent_dept_id) REFERENCES dept (dept_id),
    CONSTRAINT chk_dept_self
        CHECK (parent_dept_id IS NULL OR parent_dept_id <> dept_id)
) ENGINE = InnoDB;

CREATE TABLE emp (
    emp_id INT NOT NULL,
    dept_id INT NOT NULL,
    manager_emp_id INT NULL,
    emp_name VARCHAR(100) NOT NULL,
    email VARCHAR(200) NOT NULL,
    base_salary DECIMAL(12, 2) NOT NULL,
    profile JSON NOT NULL,
    locale_code VARCHAR(20)
        GENERATED ALWAYS AS (JSON_UNQUOTE(JSON_EXTRACT(profile, '$.locale'))) STORED,
    skill_count INT
        GENERATED ALWAYS AS (JSON_LENGTH(JSON_EXTRACT(profile, '$.skills'))) STORED,
    hired_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    PRIMARY KEY (emp_id),
    UNIQUE KEY uk_emp_email (email),
    KEY idx_emp_dept_salary (dept_id, base_salary DESC),
    KEY idx_emp_manager (manager_emp_id),
    CONSTRAINT fk_emp_dept
        FOREIGN KEY (dept_id) REFERENCES dept (dept_id),
    CONSTRAINT fk_emp_manager
        FOREIGN KEY (manager_emp_id) REFERENCES emp (emp_id),
    CONSTRAINT chk_emp_salary
        CHECK (base_salary >= 0)
) ENGINE = InnoDB;

CREATE TABLE orders (
    order_id BIGINT NOT NULL,
    emp_id INT NOT NULL,
    order_code VARCHAR(40) NOT NULL,
    order_status VARCHAR(20) NOT NULL,
    currency_code CHAR(3) NOT NULL,
    total_amount DECIMAL(14, 2) NOT NULL,
    payload JSON NOT NULL,
    created_at DATETIME(6) NOT NULL,
    customer_tier VARCHAR(20)
        GENERATED ALWAYS AS (JSON_UNQUOTE(JSON_EXTRACT(payload, '$.customer.tier'))) STORED,
    item_count INT
        GENERATED ALWAYS AS (JSON_LENGTH(JSON_EXTRACT(payload, '$.items'))) STORED,
    note_text VARCHAR(255)
        GENERATED ALWAYS AS (JSON_UNQUOTE(JSON_EXTRACT(payload, '$.note'))) VIRTUAL,
    PRIMARY KEY (order_id),
    UNIQUE KEY uk_order_code (order_code),
    KEY idx_orders_emp_created (emp_id, created_at),
    KEY idx_orders_tier_status (customer_tier, order_status),
    CONSTRAINT fk_orders_emp
        FOREIGN KEY (emp_id) REFERENCES emp (emp_id),
    CONSTRAINT chk_orders_amount
        CHECK (total_amount >= 0)
) ENGINE = InnoDB;

CREATE TABLE order_audit (
    audit_id BIGINT NOT NULL AUTO_INCREMENT,
    order_id BIGINT NOT NULL,
    action_type VARCHAR(30) NOT NULL,
    action_note VARCHAR(255) NOT NULL,
    created_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    PRIMARY KEY (audit_id),
    KEY idx_order_audit_order (order_id),
    CONSTRAINT fk_order_audit_order
        FOREIGN KEY (order_id) REFERENCES orders (order_id)
) ENGINE = InnoDB;

CREATE TABLE sales_summary (
    summary_key VARCHAR(100) NOT NULL,
    `group` VARCHAR(30) NOT NULL,
    summary_value_decimal DECIMAL(18, 2) NULL,
    summary_value_text VARCHAR(500) NULL,
    summary_json JSON NULL,
    created_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    PRIMARY KEY (summary_key)
) ENGINE = InnoDB;

CREATE TABLE error_log (
    log_id BIGINT NOT NULL AUTO_INCREMENT,
    step_name VARCHAR(100) NOT NULL,
    sql_state CHAR(5) NULL,
    mysql_errno INT NULL,
    message_text VARCHAR(512) NULL,
    created_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    PRIMARY KEY (log_id)
) ENGINE = InnoDB;

CREATE FUNCTION fn_currency_rate (p_currency_code CHAR(3))
RETURNS DECIMAL(10, 5)
DETERMINISTIC
BEGIN
    RETURN CASE UPPER(TRIM(p_currency_code))
        WHEN 'USD' THEN 1.0000
        WHEN 'EUR' THEN 1.1000
        WHEN 'KRW' THEN 0.00075
        WHEN 'JPY' THEN 0.00680
        ELSE 1.0000
    END;
END$$

CREATE PROCEDURE sp_assert (
    IN p_condition BOOLEAN,
    IN p_message VARCHAR(255)
)
BEGIN
    IF NOT p_condition THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = p_message;
    END IF;
END$$

CREATE TRIGGER trg_orders_bi
BEFORE INSERT ON orders
FOR EACH ROW
BEGIN
    SET NEW.order_status = UPPER(TRIM(NEW.order_status));
    SET NEW.currency_code = UPPER(TRIM(NEW.currency_code));

    IF NEW.total_amount < 0 THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'negative total_amount is not allowed';
    END IF;
END$$

CREATE TRIGGER trg_orders_bu
BEFORE UPDATE ON orders
FOR EACH ROW
BEGIN
    SET NEW.order_status = UPPER(TRIM(NEW.order_status));
    SET NEW.currency_code = UPPER(TRIM(NEW.currency_code));
END$$

CREATE TRIGGER trg_orders_ai
AFTER INSERT ON orders
FOR EACH ROW
BEGIN
    INSERT INTO order_audit (
        order_id,
        action_type,
        action_note
    )
    VALUES (
        NEW.order_id,
        'INSERT',
        CONCAT(
            'created tier=',
            COALESCE(JSON_UNQUOTE(JSON_EXTRACT(NEW.payload, '$.customer.tier')), 'null')
        )
    );
END$$

CREATE TRIGGER trg_orders_au
AFTER UPDATE ON orders
FOR EACH ROW
BEGIN
    IF NOT (OLD.order_status <=> NEW.order_status) THEN
        INSERT INTO order_audit (
            order_id,
            action_type,
            action_note
        )
        VALUES (
            NEW.order_id,
            'UPDATE_STATUS',
            CONCAT(OLD.order_status, ' -> ', NEW.order_status)
        );
    END IF;
END$$
