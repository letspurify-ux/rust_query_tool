CREATE TABLE cfb_user (
    user_id      INT          NOT NULL PRIMARY KEY,
    user_code    VARCHAR(30)  NOT NULL UNIQUE,
    user_name    VARCHAR(200) NOT NULL,
    status_code  CHAR(1)      NOT NULL DEFAULT 'A',
    profile_json LONGTEXT     NULL,
    note_text    LONGTEXT     NULL,
    created_at   DATETIME(6)  NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    CHECK (status_code IN ('A', 'I'))
);

CREATE TABLE cfb_order_hdr (
    order_id     BIGINT                           NOT NULL PRIMARY KEY,
    user_id      INT                              NOT NULL,
    order_no     VARCHAR(40)                      NOT NULL UNIQUE,
    order_status ENUM('NEW', 'PAID', 'CANCELLED') NOT NULL,
    ordered_at   DATETIME(6)                      NOT NULL,
    memo_text    LONGTEXT                         NULL,
    payload_json LONGTEXT                         NULL,
    created_at   DATETIME(6)                      NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    CONSTRAINT fk_cfb_order_hdr_user FOREIGN KEY (user_id) REFERENCES cfb_user(user_id)
);

CREATE TABLE cfb_item (
    item_id       BIGINT         NOT NULL PRIMARY KEY,
    order_id      BIGINT         NOT NULL,
    line_no       INT            NOT NULL,
    item_name     VARCHAR(200)   NOT NULL,
    qty           INT            NOT NULL,
    unit_price    DECIMAL(18, 2) NOT NULL,
    discount_rate DECIMAL(9, 4)  NOT NULL DEFAULT 0,
    note_text     LONGTEXT       NULL,
    created_at    DATETIME(6)    NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    CONSTRAINT uq_cfb_item UNIQUE (order_id, line_no),
    CONSTRAINT fk_cfb_item_order FOREIGN KEY (order_id) REFERENCES cfb_order_hdr(order_id),
    CHECK (qty > 0),
    CHECK (unit_price >= 0),
    CHECK (discount_rate >= 0 AND discount_rate <= 1)
);

CREATE TABLE cfb_audit (
    audit_id    BIGINT       NOT NULL AUTO_INCREMENT PRIMARY KEY,
    entity_name VARCHAR(50)  NOT NULL,
    entity_pk   VARCHAR(100) NOT NULL,
    action_name VARCHAR(50)  NOT NULL,
    detail_text LONGTEXT     NULL,
    created_at  DATETIME(6)  NOT NULL DEFAULT CURRENT_TIMESTAMP(6)
);

CREATE PROCEDURE cfb_fail(IN p_message TEXT)
BEGIN
    /* BEGIN ; END ; CASE ; DELIMITER ; */
    SET @cfb_fail_msg = LEFT(COALESCE(p_message, 'UNKNOWN FAILURE'), 128);
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = @cfb_fail_msg;
END//

CREATE PROCEDURE cfb_assert_true(IN p_condition BOOLEAN, IN p_message TEXT)
BEGIN
    -- 한 줄 주석 안의 ; 는 실행 분리 기준이 아니어야 한다 ; ; ; ;
    INSERT INTO cfb_assert_log (test_name, detail_text)
    VALUES ('ASSERT_TRUE', CONCAT('condition=', IFNULL(CAST(p_condition AS CHAR), 'NULL'), ', message=', p_message));
    IF COALESCE(p_condition, 0) = 0 THEN
        CALL cfb_fail(CONCAT('ASSERT_TRUE FAILED: ', p_message));
    END IF;
END//

CREATE PROCEDURE cfb_assert_eq_int(IN p_expected BIGINT, IN p_actual BIGINT, IN p_message TEXT)
BEGIN
    /* 주석 안 SQL 흉내
       IF 1 = 1 THEN
          SELECT 'broken';
       END IF;
    */
    INSERT INTO cfb_assert_log (test_name, detail_text)
    VALUES ('ASSERT_EQ_INT', CONCAT('expected=', p_expected, ', actual=', p_actual, ', message=', p_message));
    IF COALESCE(p_expected, - 999999999) <> COALESCE(p_actual, - 999999999) THEN
        CALL cfb_fail(
            CONCAT('ASSERT_EQ_INT FAILED: ', p_message, ' [expected=', p_expected, ', actual=', p_actual, ']')
        );
    END IF;
END//

CREATE PROCEDURE cfb_assert_eq_decimal(
    IN p_expected DECIMAL(30, 8),
    IN p_actual DECIMAL(30, 8),
    IN p_scale INT,
    IN p_message TEXT
)
BEGIN
    DECLARE v_expected DECIMAL(30, 8);
    DECLARE v_actual DECIMAL(30, 8);
    /* ;;;;; */
    SET v_expected = ROUND(COALESCE(p_expected, 0), p_scale);
    SET v_actual = ROUND(COALESCE(p_actual, 0), p_scale);
    INSERT INTO cfb_assert_log (test_name, detail_text)
    VALUES ('ASSERT_EQ_DECIMAL', CONCAT('expected=', v_expected, ', actual=', v_actual, ', scale=', p_scale, ', message=', p_message));
    IF v_expected <> v_actual THEN
        CALL cfb_fail(
            CONCAT('ASSERT_EQ_DECIMAL FAILED: ', p_message, ' [expected=', v_expected, ', actual=', v_actual, ']')
        );
    END IF;
END//

CREATE FUNCTION cfb_fn_net_amount(
    p_qty INT,
    p_unit_price DECIMAL(18, 2),
    p_discount_rate DECIMAL(9, 4)
)
RETURNS DECIMAL(18, 2)
DETERMINISTIC
BEGIN
    -- "/* not comment */" 는 문자열이 아니라 주석이다? 아니다. 여기선 그냥 주석.
    RETURN ROUND(p_qty * p_unit_price * (1 - p_discount_rate), 2);
END//

CREATE FUNCTION cfb_fn_status_name(p_status CHAR(1))
RETURNS VARCHAR(20)
DETERMINISTIC
BEGIN
    DECLARE v_name VARCHAR(20);
    SET v_name =
        CASE
            WHEN p_status = 'A' THEN
                'ACTIVE'
            WHEN p_status = 'I' THEN
                'INACTIVE'
            ELSE
                'UNKNOWN'
        END;
    RETURN v_name;
END//

CREATE PROCEDURE cfb_make_paid_order(
    IN p_order_id BIGINT,
    IN p_user_id INT,
    IN p_order_no VARCHAR(40),
    IN p_ordered_at DATETIME(6),
    IN p_item_name VARCHAR(200),
    IN p_qty INT,
    IN p_unit_price DECIMAL(18, 2),
    IN p_discount_rate DECIMAL(9, 4)
)
BEGIN
    DECLARE v_exists INT DEFAULT 0;
    /* 일부러 긴 주석
       여기 안의 세미콜론은 무시되어야 한다.
       ;
       ;
       ;
       fake END//
       fake DELIMITER ;
       fake CREATE TRIGGER ...
    */
    SELECT COUNT(*)
    INTO v_exists
    FROM cfb_user
    WHERE user_id = p_user_id
        AND status_code = 'A';
    IF v_exists = 0 THEN
        CALL cfb_fail(CONCAT('inactive or missing user: ', p_user_id));
    END IF;
    START TRANSACTION;
    INSERT INTO cfb_order_hdr (order_id, user_id, order_no, order_status, ordered_at, memo_text, payload_json)
    VALUES (p_order_id, p_user_id, p_order_no, 'PAID', p_ordered_at, 'generated by cfb_make_paid_order ; contains ''quote'' and -- text', JSON_OBJECT('source', 'proc', 'danger_text', '/* fake comment */ ; DELIMITER //'));
    INSERT INTO cfb_item (item_id, order_id, line_no, item_name, qty, unit_price, discount_rate, note_text)
    VALUES (p_order_id * 10, p_order_id, 1, p_item_name, p_qty, p_unit_price, p_discount_rate, 'generated item note ; -- /* */');
    COMMIT;
END//

CREATE PROCEDURE cfb_build_temp_report()
BEGIN
    DECLARE v_sql LONGTEXT;
    /* 동적 SQL 텍스트 안에 주석 문자처럼 보이는 것 넣기 */
    SET v_sql = '
        CREATE TEMPORARY TABLE tmp_cfb_report AS
        SELECT
            oh.user_id,
            DATE_FORMAT(oh.ordered_at, ''%Y-%m'') AS ym,
            COUNT(DISTINCT oh.order_id) AS order_count,
            ROUND(SUM(cfb_fn_net_amount(i.qty, i.unit_price, i.discount_rate)), 2) AS net_amount
        FROM cfb_order_hdr oh
        JOIN cfb_item i
          ON i.order_id = oh.order_id
        WHERE oh.order_status = ''PAID''
        GROUP BY oh.user_id, DATE_FORMAT(oh.ordered_at, ''%Y-%m'')
    ';
    PREPARE stmt
    FROM v_sql;
    EXECUTE stmt;
    DEALLOCATE PREPARE stmt;
END//

CREATE OR REPLACE VIEW cfb_v_result AS
    SELECT u.user_id,
        u.user_code,
        u.user_name,
        cfb_fn_status_name(u.status_code) AS user_status_name,
        COALESCE(x.paid_order_count, 0) AS paid_order_count,
        COALESCE(x.total_net_amount, 0.00) AS total_net_amount,
    CASE
        WHEN COALESCE(x.total_net_amount, 0) >= 1500 THEN 'TOP'
        WHEN COALESCE(x.total_net_amount, 0) >= 500 THEN 'MID'
        WHEN COALESCE(x.total_net_amount, 0) > 0 THEN 'LOW'
        ELSE 'NONE'
    END AS spend_band
    FROM cfb_user u
    LEFT JOIN (
        SELECT oh.user_id,
            COUNT(DISTINCT oh.order_id) AS paid_order_count,
            ROUND(SUM(cfb_fn_net_amount(i.qty, i.unit_price, i.discount_rate)), 2) AS total_net_amount
        FROM cfb_order_hdr oh
        JOIN cfb_item i
            ON i.order_id = oh.order_id
        WHERE oh.order_status = 'PAID'
        GROUP BY oh.user_id
    ) x
        ON x.user_id = u.user_id;

WITH paid_line AS (
    SELECT
        oh.user_id,
        oh.order_id,
        i.item_id,
        cfb_fn_net_amount(i.qty, i.unit_price, i.discount_rate) AS net_amount
    FROM cfb_order_hdr oh
    JOIN cfb_item i
        ON i.order_id = oh.order_id
    WHERE oh.order_status = 'PAID'
),
ranked AS (
    SELECT
        p.*,
        ROW_NUMBER() OVER (
            PARTITION BY p.user_id
            ORDER BY p.net_amount DESC,
            p.item_id
        ) AS rn
    FROM paid_line p
)
SELECT COUNT(*)
INTO @top_line_count
FROM ranked
WHERE rn = 1;

CALL cfb_assert_eq_int(
    2,
    @top_line_count,
    'top line count for users with paid orders'
);
