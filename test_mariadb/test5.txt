CREATE OR REPLACE VIEW v_item_rollup AS
    SELECT wi.item_id,
        wi.project_id,
        wi.sprint_id,
        wi.assignee_user_id,
        wi.item_code,
        wi.status,
        wi.points,
        COUNT(we.event_id) AS event_count,
        ROUND(COALESCE(SUM(we.delta_hours), 0), 2) AS total_hours,
        MAX(we.event_at) AS last_event_at
    FROM work_item wi
    LEFT JOIN work_event we
        ON we.item_id = wi.item_id
    GROUP BY wi.item_id,
        wi.project_id,
        wi.sprint_id,
        wi.assignee_user_id,
        wi.item_code,
        wi.status,
        wi.points;

DELIMITER $$

CREATE TRIGGER bi_work_item
    BEFORE INSERT
    ON work_item
    FOR EACH ROW
BEGIN
    IF NEW.STATUS NOT IN ('NEW', 'IN_PROGRESS', 'REVIEW', 'DONE', 'BLOCKED') THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'invalid work_item.status';
    END IF;
    IF NEW.points IS NULL
            OR NEW.points <= 0 THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'work_item.points must be > 0';
    END IF;
    IF NEW.detail_doc IS NULL
            OR JSON_VALID(NEW.detail_doc) = 0 THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'work_item.detail_doc must be valid JSON';
    END IF;
    SET NEW.title = TRIM(NEW.title);
    IF CHAR_LENGTH(NEW.title) = 0 THEN
        SET NEW.title = CONCAT('UNTITLED-', NEW.item_code);
    END IF;
    SET NEW.created_at = COALESCE(NEW.created_at, CURRENT_TIMESTAMP(6));
    SET NEW.updated_at = COALESCE(NEW.updated_at, NEW.created_at);
END$$

CREATE TRIGGER bu_work_item
    BEFORE UPDATE
    ON work_item
    FOR EACH ROW
BEGIN
    IF NEW.STATUS NOT IN ('NEW', 'IN_PROGRESS', 'REVIEW', 'DONE', 'BLOCKED') THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'invalid work_item.status on update';
    END IF;
    IF NEW.detail_doc IS NULL
            OR JSON_VALID(NEW.detail_doc) = 0 THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'work_item.detail_doc must remain valid JSON';
    END IF;
    SET NEW.updated_at = CURRENT_TIMESTAMP(6);
    SET NEW.revision_no = OLD.revision_no + 1;
END$$

CREATE TRIGGER ai_work_item
    AFTER INSERT
    ON work_item
    FOR EACH ROW
BEGIN
    INSERT INTO audit_log (entity_name, entity_id, action_name, payload)
    VALUES ('work_item', NEW.item_id, 'WORK_ITEM_INSERT', JSON_OBJECT('item_code', NEW.item_code, 'status', NEW.status, 'points', NEW.points, 'revision_no', NEW.revision_no));
END$$

CREATE TRIGGER au_work_item
    AFTER UPDATE
    ON work_item
    FOR EACH ROW
BEGIN
    INSERT INTO audit_log (entity_name, entity_id, action_name, payload)
    VALUES ('work_item', NEW.item_id, 'WORK_ITEM_UPDATE', JSON_OBJECT('item_code', NEW.item_code, 'old_status', OLD.status, 'new_status', NEW.status, 'old_revision_no', OLD.revision_no, 'new_revision_no', NEW.revision_no, 'updated_at', DATE_FORMAT(NEW.updated_at, '%Y-%m-%d %H:%i:%s.%f')));
END$$

CREATE TRIGGER bi_work_event
    BEFORE INSERT
    ON work_event
    FOR EACH ROW
BEGIN
    IF NEW.event_type NOT IN ('CREATED', 'STARTED', 'PROGRESSED', 'REVIEWED', 'CLOSED') THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'invalid work_event.event_type';
    END IF;
    IF NEW.delta_hours IS NULL
            OR NEW.delta_hours < 0
            OR NEW.delta_hours > 24 THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'work_event.delta_hours must be between 0 and 24';
    END IF;
    IF NEW.event_doc IS NULL
            OR JSON_VALID(NEW.event_doc) = 0 THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'work_event.event_doc must be valid JSON';
    END IF;
    IF NEW.note IS NULL
            OR CHAR_LENGTH(TRIM(NEW.note)) = 0 THEN
        SET NEW.note = CONCAT('auto-note:', NEW.actor_user_id, ':', DATE_FORMAT(NEW.event_at, '%Y%m%d%H%i%s'));
    END IF;
END$$

CREATE TRIGGER ai_work_event
    AFTER INSERT
    ON work_event
    FOR EACH ROW
BEGIN
    INSERT INTO audit_log (entity_name, entity_id, action_name, payload)
    VALUES ('work_event', NEW.event_id, 'WORK_EVENT_INSERT', JSON_OBJECT('item_id', NEW.item_id, 'actor_user_id', NEW.actor_user_id, 'event_type', NEW.event_type, 'delta_hours', NEW.delta_hours, 'event_at', DATE_FORMAT(NEW.event_at, '%Y-%m-%d %H:%i:%s.%f'), 'note', NEW.note));
END$$

CREATE FUNCTION fn_complexity_band(
    p_points INT,
    p_hours DECIMAL(12, 2)
)
RETURNS VARCHAR(20)
DETERMINISTIC
BEGIN
    DECLARE v_score DECIMAL(12, 4);
    SET v_score = IFNULL(p_hours / NULLIF(p_points, 0), 0);
    RETURN
    CASE
        WHEN v_score >= 2.50 THEN
            'critical'
        WHEN v_score >= 1.25 THEN
            'watch'
        ELSE
            'ok'
    END;
END$$

CREATE PROCEDURE sp_assert_eq_bigint(
    IN p_expected BIGINT,
    IN p_actual BIGINT,
    IN p_message VARCHAR(255)
)
BEGIN
    IF NOT (p_expected <=> p_actual) THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = p_message;
    END IF;
END//

CREATE PROCEDURE sp_assert_eq_decimal(
    IN p_expected DECIMAL(20, 6),
    IN p_actual DECIMAL(20, 6),
    IN p_scale INT,
    IN p_message VARCHAR(255)
)
BEGIN
    IF ROUND(IFNULL(p_expected, 0), p_scale) <> ROUND(IFNULL(p_actual, 0), p_scale) THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = p_message;
    END IF;
END//

CREATE PROCEDURE sp_fault_injection_handled()
BEGIN
    DECLARE v_step VARCHAR(50) DEFAULT NULL;
    DECLARE v_sqlstate CHAR(5);
    DECLARE v_errno INT;
    DECLARE v_message TEXT;
    DECLARE user_error CONDITION
    FOR SQLSTATE '45000';
    DECLARE CONTINUE HANDLER FOR user_error
        BEGIN
            GET DIAGNOSTICS CONDITION 1 v_sqlstate = RETURNED_SQLSTATE,
            v_errno = MYSQL_ERRNO,
            v_message = MESSAGE_TEXT;
            INSERT INTO qa_error_capture (phase_name, sqlstate_char, mysql_errno, message_text, ctx)
            VALUES ('fault_injection', v_sqlstate, v_errno, LEFT(v_message, 512), JSON_OBJECT('step', v_step));
        END;
    SET v_step = 'bad_work_item_json';
    INSERT INTO work_item (project_id, sprint_id, assignee_user_id, reporter_user_id, item_code, title, status, points, detail_doc)
    VALUES (1, 1, 3, 2, 'BAD-JSON-001', 'broken item', 'NEW', 3, 'not valid json');
    SET v_step = 'bad_work_item_status_update';
    UPDATE work_item
    SET status = 'NOT_A_STATUS'
    WHERE item_id = 1;
    SET v_step = 'bad_work_event_type_and_hours';
    INSERT INTO work_event (item_id, actor_user_id, event_type, event_at, delta_hours, note, event_doc)
    VALUES (1, 3, 'BROKEN_EVENT', '2025-01-02 11:00:00.000000', 99.99, NULL, '{]');
END//

CREATE PROCEDURE sp_upsert_one_month_stat(
    IN p_ym CHAR(7),
    IN p_project_id INT
)
BEGIN
    INSERT INTO monthly_project_stats (ym, project_id, item_count, event_count, total_hours, avg_item_hours, completed_items, blocked_items, top_actor_user_id, anomaly_count, rebuilt_at)
    SELECT t.ym,
        t.project_id,
        t.item_count,
        t.event_count,
        t.total_hours,
        t.avg_item_hours,
        t.completed_items,
        t.blocked_items,
        t.top_actor_user_id,
        t.anomaly_count,
        CURRENT_TIMESTAMP(6)
    FROM tmp_monthly_stats t
    WHERE t.ym = p_ym
        AND t.project_id = p_project_id
        ON DUPLICATE KEY UPDATE item_count = VALUES(item_count),
        event_count = VALUES(event_count),
        total_hours = VALUES(total_hours),
        avg_item_hours = VALUES(avg_item_hours),
        completed_items = VALUES(completed_items),
        blocked_items = VALUES(blocked_items),
        top_actor_user_id = VALUES(top_actor_user_id),
        anomaly_count = VALUES(anomaly_count),
        rebuilt_at = VALUES(rebuilt_at);
END//

CREATE PROCEDURE sp_rebuild_monthly_stats(
    IN p_from_ym CHAR(7),
    IN p_to_ym CHAR(7)
)
BEGIN
    DECLARE v_done INT DEFAULT 0;
    DECLARE v_ym CHAR(7);
    DECLARE v_project_id INT;
    DECLARE v_project_code VARCHAR(30);
    DECLARE cur CURSOR FOR
        SELECT t.ym,
        t.project_id,
        p.project_code
        FROM tmp_monthly_stats t
        JOIN project p
            ON p.project_id = t.project_id
        ORDER BY t.ym,
            t.project_id;
    DECLARE CONTINUE HANDLER FOR NOT FOUND
        SET v_done = 1;
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
        BEGIN
            ROLLBACK;
            RESIGNAL;
        END;
    DROP TEMPORARY TABLE IF EXISTS tmp_monthly_stats;
    SET @SQL = CONCAT('CREATE TEMPORARY TABLE tmp_monthly_stats AS ', 'SELECT m.ym, m.project_id, m.item_count, m.event_count, m.total_hours, m.avg_item_hours, ', '       m.completed_items, m.blocked_items, at.top_actor_user_id, m.anomaly_count ', 'FROM (', '    SELECT DATE_FORMAT(we.event_at, ''%Y-%m'') AS ym, wi.project_id, ', '           COUNT(DISTINCT wi.item_id) AS item_count, ', '           COUNT(*) AS event_count, ', '           ROUND(SUM(we.delta_hours), 2) AS total_hours, ', '           ROUND(SUM(we.delta_hours) / NULLIF(COUNT(DISTINCT wi.item_id), 0), 2) AS avg_item_hours, ', '           COUNT(DISTINCT CASE WHEN wi.status = ''DONE'' THEN wi.item_id END) AS completed_items, ', '           COUNT(DISTINCT CASE WHEN wi.status = ''BLOCKED'' THEN wi.item_id END) AS blocked_items, ', '           COUNT(DISTINCT CASE WHEN we.delta_hours >= 6 THEN wi.item_id END) AS anomaly_count ', '    FROM work_event we ', '    JOIN work_item wi ON wi.item_id = we.item_id ', '    WHERE DATE_FORMAT(we.event_at, ''%Y-%m'') BETWEEN ', QUOTE(p_from_ym), ' AND ', QUOTE(p_to_ym), ' ', '    GROUP BY DATE_FORMAT(we.event_at, ''%Y-%m''), wi.project_id', ') m ', 'LEFT JOIN (', '    SELECT ranked.ym, ranked.project_id, ranked.actor_user_id AS top_actor_user_id ', '    FROM (', '        SELECT x.ym, x.project_id, x.actor_user_id, x.actor_hours, ', '               ROW_NUMBER() OVER (PARTITION BY x.ym, x.project_id ORDER BY x.actor_hours DESC, x.actor_user_id) AS rn ', '        FROM (', '            SELECT DATE_FORMAT(we.event_at, ''%Y-%m'') AS ym, wi.project_id, we.actor_user_id, ROUND(SUM(we.delta_hours), 2) AS actor_hours ', '            FROM work_event we ', '            JOIN work_item wi ON wi.item_id = we.item_id ', '            WHERE DATE_FORMAT(we.event_at, ''%Y-%m'') BETWEEN ', QUOTE(p_from_ym), ' AND ', QUOTE(p_to_ym), ' ', '            GROUP BY DATE_FORMAT(we.event_at, ''%Y-%m''), wi.project_id, we.actor_user_id', '        ) x', '    ) ranked ', '    WHERE ranked.rn = 1', ') at ON at.ym = m.ym AND at.project_id = m.project_id');
    PREPARE stmt
    FROM @sql;
    EXECUTE stmt;
    DEALLOCATE PREPARE stmt;
    START TRANSACTION;
    OPEN cur;
    read_loop: LOOP
        FETCH cur
        INTO v_ym,
            v_project_id,
            v_project_code;
        IF v_done = 1 THEN
            LEAVE read_loop;
        END IF;
        SAVEPOINT sp_one;
        BEGIN
            DECLARE inner_sqlstate CHAR(5);
            DECLARE inner_errno INT;
            DECLARE inner_message TEXT;
            DECLARE EXIT HANDLER FOR SQLEXCEPTION
                BEGIN
                    GET DIAGNOSTICS CONDITION 1 inner_sqlstate = RETURNED_SQLSTATE,
                    inner_errno = MYSQL_ERRNO,
                    inner_message = MESSAGE_TEXT;
                    ROLLBACK TO SAVEPOINT sp_one;
                    INSERT INTO qa_error_capture (phase_name, sqlstate_char, mysql_errno, message_text, ctx)
                    VALUES ('rollup', inner_sqlstate, inner_errno, LEFT(inner_message, 512), JSON_OBJECT('ym', v_ym, 'project_code', v_project_code, 'path', 'savepoint-recovery'));
                    CALL sp_upsert_one_month_stat(v_ym, v_project_id);
                END;
            IF v_project_code = 'CHAOS'
                    AND v_ym = '2025-02' THEN
                INSERT INTO work_event (item_id, actor_user_id, event_type, event_at, delta_hours, note, event_doc)
                VALUES (
                    (
                        SELECT MIN(item_id)
                        FROM work_item
                        WHERE project_id = v_project_id
                    ),
                        6,
                        'BROKEN_EVENT',
                        '2025-02-10 10:00:00.000000',
                        999.00,
                        NULL,
                        'not json'
                );
            END IF;
            CALL sp_upsert_one_month_stat(v_ym, v_project_id);
        END;
        RELEASE SAVEPOINT sp_one;
    END LOOP;
    CLOSE cur;
    COMMIT;
    DROP TEMPORARY TABLE IF EXISTS tmp_monthly_stats;
END//

CREATE PROCEDURE sp_emit_result_sets()
BEGIN
    SELECT
        (
        SELECT COUNT(*)
        FROM org_unit
        ) AS org_unit_count,
        (
        SELECT COUNT(*)
        FROM app_user
        ) AS app_user_count,
        (
        SELECT COUNT(*)
        FROM project
        ) AS project_count,
        (
        SELECT COUNT(*)
        FROM sprint
        ) AS sprint_count,
        (
        SELECT COUNT(*)
        FROM work_item
        ) AS work_item_count,
        (
        SELECT COUNT(*)
        FROM work_event
        ) AS work_event_count,
        (
        SELECT COUNT(*)
        FROM monthly_project_stats
        ) AS monthly_project_stats_count,
        (
        SELECT COUNT(*)
        FROM qa_error_capture
        ) AS qa_error_capture_count,
        (
        SELECT COUNT(*)
        FROM audit_log
        ) AS audit_log_count;
    SELECT capture_id,
        phase_name,
        sqlstate_char,
        mysql_errno,
        message_text,
        ctx,
        created_at
    FROM qa_error_capture
    ORDER BY capture_id;
    SELECT m.ym,
        p.project_code,
        m.item_count,
        m.event_count,
        m.total_hours,
        m.avg_item_hours,
        m.completed_items,
        m.blocked_items,
        m.top_actor_user_id,
        m.anomaly_count,
        m.rebuilt_at
    FROM monthly_project_stats m
    JOIN project p
        ON p.project_id = m.project_id
    ORDER BY m.ym,
        p.project_code;
END//

-- ============================================================
-- 8) Monster query #1
--    recursive org tree + JSON_TABLE + member rollup
-- ============================================================

WITH RECURSIVE org_tree AS (
    SELECT
        ou.org_unit_id,
        ou.parent_org_unit_id,
        ou.unit_code,
        ou.unit_name,
        CAST(ou.unit_name AS CHAR(1000)) AS unit_path,
        1 AS lvl
    FROM org_unit ou
    WHERE ou.parent_org_unit_id IS NULL
    UNION ALL
    SELECT
        c.org_unit_id,
        c.parent_org_unit_id,
        c.unit_code,
        c.unit_name,
        CONCAT(t.unit_path, ' > ', c.unit_name) AS unit_path,
        t.lvl + 1 AS lvl
    FROM org_unit c
    JOIN org_tree t
        ON t.org_unit_id = c.parent_org_unit_id
),
member_rollup AS (
    SELECT
        pm.project_id,
        GROUP_CONCAT(CONCAT(u.user_code, ':', pm.role_name, '@', pm.allocation_pct) ORDER BY pm.allocation_pct DESC, u.user_code SEPARATOR ' | ') AS members
    FROM project_member pm
    JOIN app_user u
        ON u.user_id = pm.user_id
    GROUP BY pm.project_id
)
SELECT
    p.project_code,
    p.project_name,
    owner.display_name AS owner_name,
    ot.unit_path AS owner_unit_path,
    mr.members,
    jt.label AS spec_label
FROM project p
JOIN app_user owner
    ON owner.user_id = p.owner_user_id
JOIN org_tree ot
    ON ot.org_unit_id = owner.org_unit_id
JOIN member_rollup mr
    ON mr.project_id = p.project_id
CROSS JOIN JSON_TABLE(p.spec_json, '$.labels[*]' COLUMNS(
    label VARCHAR(40) PATH '$'
)) AS jt
ORDER BY p.project_code,
    jt.label;

-- ============================================================
-- 9) Monster query #2
--    recursive calendar + window + correlated subquery + function
-- ============================================================

WITH RECURSIVE month_dim AS (
    SELECT CAST('2025-01-01' AS DATE) AS month_start
    UNION ALL
    SELECT DATE_ADD(month_start, INTERVAL 1 MONTH)
    FROM month_dim
    WHERE month_start < '2025-03-01'
),
daily_project AS (
    SELECT
        DATE(we.event_at) AS event_day,
        wi.project_id,
        p.project_code,
        ROUND(SUM(we.delta_hours), 2) AS day_hours,
        COUNT(*) AS event_count
    FROM work_event we
    JOIN work_item wi
        ON wi.item_id = we.item_id
    JOIN project p
        ON p.project_id = wi.project_id
    GROUP BY DATE(we.event_at),
        wi.project_id,
        p.project_code
),
calendarized AS (
    SELECT
        md.month_start,
        dp.project_id,
        dp.project_code,
        dp.event_day,
        dp.day_hours,
        dp.event_count
    FROM month_dim md
    LEFT JOIN daily_project dp
        ON DATE_FORMAT(dp.event_day, '%Y-%m-01') = DATE_FORMAT(md.month_start, '%Y-%m-01')
),
ranked AS (
    SELECT
        c.*,
    ROW_NUMBER() OVER (
        PARTITION BY c.project_id,
            DATE_FORMAT(c.event_day, '%Y-%m')
        ORDER BY c.day_hours DESC,
                c.event_day
    ) AS rn,
    DENSE_RANK() OVER (
        PARTITION BY c.project_id,
            DATE_FORMAT(c.event_day, '%Y-%m')
        ORDER BY c.day_hours DESC
    ) AS dense_rnk,
    SUM(c.day_hours) OVER (
        PARTITION BY c.project_id,
            DATE_FORMAT(c.event_day, '%Y-%m')
        ORDER BY c.event_day
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS running_hours,
    AVG(c.day_hours) OVER (
        PARTITION BY c.project_id
        ORDER BY c.event_day
        ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
    ) AS moving_avg_3d,
    LAG(c.day_hours) OVER (
        PARTITION BY c.project_id
        ORDER BY c.event_day
    ) AS prev_hours
    FROM calendarized c
    WHERE c.event_day IS NOT NULL
)
SELECT
    r.project_code,
    DATE_FORMAT(r.event_day, '%Y-%m') AS ym,
    r.event_day,
    r.day_hours,
    r.event_count,
    r.rn,
    r.dense_rnk,
    ROUND(r.running_hours, 2) AS running_hours,
    ROUND(r.moving_avg_3d, 2) AS moving_avg_3d,
    COALESCE(r.prev_hours, 0) AS prev_hours,
    fn_complexity_band(
        (
            SELECT ROUND(AVG(wi.points), 0)
            FROM work_item wi
            WHERE wi.project_id = r.project_id
        ),
        r.running_hours) AS risk_band,
    (
        SELECT ROUND(AVG(we2.delta_hours), 2)
        FROM work_event we2
        JOIN work_item wi2
            ON wi2.item_id = we2.item_id
        WHERE wi2.project_id = r.project_id
            AND we2.event_at BETWEEN DATE_SUB(r.event_day, INTERVAL 7 DAY) AND DATE_ADD(r.event_day, INTERVAL 1 DAY)
    ) AS trailing_7d_avg
FROM ranked r
WHERE r.rn <= 3
ORDER BY r.project_code,
    ym,
    r.rn,
    r.event_day;

-- ============================================================
-- 10) Monster query #3
--     JSON_TABLE + window rank + correlated subquery
-- ============================================================

WITH component_usage AS (
    SELECT
        p.project_code,
        comp.component_name,
        COUNT(*) AS item_hits,
        ROUND(SUM(v.total_hours), 2) AS total_hours,
        ROUND(AVG(v.total_hours), 2) AS avg_item_hours,
        SUM(
            CASE
                WHEN wi.status = 'BLOCKED' THEN 1
                ELSE 0
            END
        ) AS blocked_hits
    FROM work_item wi
    JOIN project p
        ON p.project_id = wi.project_id
    JOIN v_item_rollup v
        ON v.item_id = wi.item_id
    CROSS JOIN JSON_TABLE(wi.detail_doc, '$.components[*]' COLUMNS(
        component_name VARCHAR(40) PATH '$'
    )) AS comp
    GROUP BY p.project_code,
        comp.component_name
),
ranked AS (
    SELECT
        cu.*,
    ROW_NUMBER() OVER (
        PARTITION BY cu.project_code
        ORDER BY cu.item_hits DESC,
                cu.total_hours DESC,
            cu.component_name
    ) AS component_rank,
    SUM(cu.total_hours) OVER (
        PARTITION BY cu.project_code
        ORDER BY cu.total_hours DESC,
                cu.component_name
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS cumulative_component_hours
    FROM component_usage cu
)
SELECT
    r.project_code,
    r.component_name,
    r.item_hits,
    r.total_hours,
    r.avg_item_hours,
    r.blocked_hits,
    r.component_rank,
    ROUND(r.cumulative_component_hours, 2) AS cumulative_component_hours,
    (
        SELECT COUNT(*)
        FROM work_item wi2
        CROSS JOIN JSON_TABLE(wi2.detail_doc, '$.labels[*]' COLUMNS(
            label_name VARCHAR(40) PATH '$'
        )) AS lab
        WHERE wi2.project_id = (
            SELECT p2.project_id
            FROM project p2
            WHERE p2.project_code = r.project_code
        )
            AND lab.label_name = CONCAT('s', r.component_rank)
    ) AS matching_rank_label_items
FROM ranked r
ORDER BY r.project_code,
    r.component_rank,
    r.component_name;

-- 11) Final inspection
-- ============================================================

SELECT m.ym,
    p.project_code,
    m.item_count,
    m.event_count,
    m.total_hours,
    m.avg_item_hours,
    m.completed_items,
    m.blocked_items,
    u.user_code AS top_actor_code,
    m.anomaly_count,
    m.rebuilt_at
FROM monthly_project_stats m
JOIN project p
    ON p.project_id = m.project_id
LEFT JOIN app_user u
    ON u.user_id = m.top_actor_user_id
ORDER BY m.ym,
    p.project_code;
