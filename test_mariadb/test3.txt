CREATE TABLE stage_node (
    node_id INT NOT NULL,
    parent_node_id INT NULL,
    `group` VARCHAR(20) NOT NULL,
    node_name VARCHAR(100) NOT NULL,
    sort_no INT NOT NULL,
    meta JSON NOT NULL,
    path_hint VARCHAR(300)
        GENERATED ALWAYS AS (JSON_UNQUOTE(JSON_EXTRACT(meta, '$.pathHint'))) VIRTUAL,
    PRIMARY KEY (node_id),
    UNIQUE KEY uk_stage_node_group_name (`group`, node_name),
    KEY idx_stage_node_parent (parent_node_id),
    CONSTRAINT fk_stage_node_parent
        FOREIGN KEY (parent_node_id) REFERENCES stage_node (node_id),
    CONSTRAINT chk_stage_node_self
        CHECK (parent_node_id IS NULL OR parent_node_id <> node_id)
) ENGINE = InnoDB;

CREATE TABLE run_case (
    run_id BIGINT NOT NULL,
    node_id INT NOT NULL,
    owner_name VARCHAR(100) NOT NULL,
    status_code VARCHAR(20) NOT NULL,
    priority INT NOT NULL,
    `rank` INT NOT NULL,
    payload JSON NOT NULL,
    created_at DATETIME(6) NOT NULL,
    locale_code VARCHAR(20)
        GENERATED ALWAYS AS (JSON_UNQUOTE(JSON_EXTRACT(payload, '$.locale'))) STORED,
    step_count INT
        GENERATED ALWAYS AS (JSON_LENGTH(JSON_EXTRACT(payload, '$.steps'))) STORED,
    note_text VARCHAR(255)
        GENERATED ALWAYS AS (JSON_UNQUOTE(JSON_EXTRACT(payload, '$.note'))) VIRTUAL,
    PRIMARY KEY (run_id),
    UNIQUE KEY uk_run_case_owner_rank (owner_name, `rank`),
    KEY idx_run_case_node_status (node_id, status_code),
    KEY idx_run_case_owner_priority (owner_name, priority DESC),
    CONSTRAINT fk_run_case_node
        FOREIGN KEY (node_id) REFERENCES stage_node (node_id),
    CONSTRAINT chk_run_case_priority
        CHECK (priority BETWEEN 1 AND 9)
) ENGINE = InnoDB;

CREATE TABLE run_audit (
    audit_id BIGINT NOT NULL AUTO_INCREMENT,
    run_id BIGINT NOT NULL,
    audit_type VARCHAR(30) NOT NULL,
    old_status VARCHAR(20) NULL,
    new_status VARCHAR(20) NULL,
    audit_note VARCHAR(255) NOT NULL,
    created_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    PRIMARY KEY (audit_id),
    KEY idx_run_audit_run (run_id)
) ENGINE = InnoDB;

CREATE TABLE qa_summary (

CREATE TABLE error_log (

CREATE FUNCTION fn_priority_factor (p_priority INT)
RETURNS DECIMAL(10, 2)
DETERMINISTIC
BEGIN
    RETURN CASE
        WHEN p_priority >= 9 THEN 4.00
        WHEN p_priority >= 8 THEN 3.00
        WHEN p_priority >= 7 THEN 2.50
        WHEN p_priority >= 6 THEN 2.00
        WHEN p_priority >= 5 THEN 1.50
        ELSE 1.00
    END;
END$$

CREATE PROCEDURE sp_assert (
    IN p_condition BOOLEAN,
    IN p_message VARCHAR(255)
)
BEGIN
    IF NOT p_condition THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = p_message;
    END IF;
END$$

CREATE PROCEDURE sp_shift_rank (
    INOUT p_rank INT,
    IN p_delta INT
)
BEGIN
    SET p_rank = p_rank + p_delta;
END$$

CREATE PROCEDURE sp_collect_status_counts (
    OUT p_done_cnt INT,
    OUT p_running_cnt INT,
    OUT p_other_cnt INT
)
BEGIN
    SELECT
        SUM(CASE WHEN status_code = 'DONE' THEN 1 ELSE 0 END),
        SUM(CASE WHEN status_code = 'RUNNING' THEN 1 ELSE 0 END),
        SUM(CASE WHEN status_code NOT IN ('DONE', 'RUNNING') THEN 1 ELSE 0 END)
    INTO
        p_done_cnt,
        p_running_cnt,
        p_other_cnt
    FROM run_case;
END$$

CREATE TRIGGER trg_run_case_bi
BEFORE INSERT ON run_case
FOR EACH ROW
BEGIN
    SET NEW.owner_name = LOWER(TRIM(NEW.owner_name));
    SET NEW.status_code = UPPER(TRIM(NEW.status_code));

    IF NEW.priority NOT BETWEEN 1 AND 9 THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'priority must be between 1 and 9';
    END IF;
END$$

CREATE TRIGGER trg_run_case_bu
BEFORE UPDATE ON run_case
FOR EACH ROW
BEGIN
    SET NEW.owner_name = LOWER(TRIM(NEW.owner_name));
    SET NEW.status_code = UPPER(TRIM(NEW.status_code));
END$$

CREATE TRIGGER trg_run_case_ai
AFTER INSERT ON run_case

BEGIN
    INSERT INTO run_audit (
        run_id,

            'inserted step_count=',

END$$

CREATE TRIGGER trg_run_case_au
AFTER UPDATE ON run_case
FOR EACH ROW
BEGIN
    IF NOT (OLD.status_code <=> NEW.status_code) THEN
        INSERT INTO run_audit (
            run_id,
            audit_type,
            old_status,
            new_status,
            audit_note
        )
        VALUES (
            NEW.run_id,
            'STATUS_CHANGE',
            OLD.status_code,
            NEW.status_code,
            CONCAT(OLD.status_code, ' -> ', NEW.status_code)
        );
    END IF;
END$$

            rm.owner_name,
            rm.created_at,
            rm.priority,
            rm.total_minutes,
            fn_priority_factor(rm.priority) AS priority_factor,
            ROUND(rm.total_minutes * fn_priority_factor(rm.priority), 2) AS weighted_minutes
        FROM run_minutes AS rm
    ),
    ranked AS (
        SELECT
            s.*,
            ROW_NUMBER() OVER w_owner AS rn_in_owner,
            LAG(s.weighted_minutes) OVER w_owner AS prev_weighted_minutes,
            SUM(s.weighted_minutes) OVER w_owner_running AS running_owner_weighted,
            DENSE_RANK() OVER w_global AS global_rank
        FROM scored AS s
        WINDOW
            w_owner AS (
                PARTITION BY s.owner_name
                ORDER BY s.created_at, s.run_id
            ),
            w_owner_running AS (
                PARTITION BY s.owner_name
                ORDER BY s.created_at, s.run_id
                ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
            ),
            w_global AS (
                ORDER BY s.weighted_minutes DESC, s.owner_name, s.run_id
            )
    )
    SELECT MAX(running_owner_weighted)
      INTO v_alice_running_weighted
      FROM ranked
     WHERE owner_name = 'alice';

    CALL sp_assert(v_alice_running_weighted = 157.50, 'window running weighted sum mismatch');

    WITH run_minutes AS (
        SELECT
            a.run_id,
            a.owner_name,
            r.created_at,
            r.priority,
            ROUND(SUM(a.minutes), 2) AS total_minutes
        FROM tmp_step_agg AS a
        JOIN run_case AS r
          ON r.run_id = a.run_id
        GROUP BY
            a.run_id,
            a.owner_name,
            r.created_at,
            r.priority
    ),
    owner_rollup AS (
        SELECT
            owner_name,
            ROUND(SUM(total_minutes * fn_priority_factor(priority)), 2) AS owner_weighted
        FROM run_minutes
        GROUP BY owner_name
    ),
    owner_ranked AS (
        SELECT
            owner_name,
            owner_weighted,
            ROW_NUMBER() OVER (
                ORDER BY owner_weighted DESC, owner_name
            ) AS rn
        FROM owner_rollup
    )
    SELECT
        owner_name,
        owner_weighted
    INTO
        v_top_owner,
        v_top_owner_weighted
    FROM owner_ranked
    WHERE rn = 1;

    CALL sp_assert(v_top_owner = 'alice', 'top owner mismatch');
    CALL sp_assert(v_top_owner_weighted = 157.50, 'top owner weighted mismatch');

    WITH run_minutes AS (
        SELECT
            a.run_id,
            a.owner_name,
            r.created_at,
            r.priority,
            ROUND(SUM(a.minutes), 2) AS total_minutes
        FROM tmp_step_agg AS a
        JOIN run_case AS r
          ON r.run_id = a.run_id
        GROUP BY
            a.run_id,
            a.owner_name,
            r.created_at,
            r.priority
    ),
    scored AS (
        SELECT
            rm.run_id,
            ROUND(rm.total_minutes * fn_priority_factor(rm.priority), 2) AS weighted_minutes
        FROM run_minutes AS rm
    ),
    ranked AS (
        SELECT
            run_id,
            weighted_minutes,
            ROW_NUMBER() OVER (
                ORDER BY weighted_minutes DESC, run_id
            ) AS rn
        FROM scored
    )
    SELECT run_id
      INTO v_top_run_id
      FROM ranked
     WHERE rn = 1;

    CALL sp_assert(v_top_run_id = 4001, 'top run id mismatch');

    INSERT INTO qa_summary (
        summary_key,
        `group`,
        `rank`,
        summary_num,
        summary_text,
        summary_json
    )
    VALUES
        (
            'TOP_OWNER_WEIGHTED',
            'METRIC',
            1,
            v_top_owner_weighted,
            v_top_owner,
            JSON_OBJECT(
                'owner', v_top_owner,
                'weighted', v_top_owner_weighted
            )
        ),
        (
            'ALICE_RUNNING_WEIGHTED',
            'METRIC',
            2,
            v_alice_running_weighted,
            'running owner weighted max',
            JSON_OBJECT(
                'owner', 'alice',
                'value', v_alice_running_weighted
            )
        ),
        (
            'TREE_DEPTH',
            'METRIC',
            3,
            v_tree_depth,
            'recursive cte max depth',
            JSON_OBJECT('depth', v_tree_depth)
        ),
        (
            'STEP_TAG_ROWS',
            'METRIC',
            4,
            v_total_step_rows,
            'json_table nested tag rows',
            JSON_OBJECT('rows', v_total_step_rows)
        ),
        (
            'STEP_TOTAL_MINUTES',
            'METRIC',
            5,
            v_total_step_minutes,
            'sum of distinct step minutes',
            JSON_OBJECT('minutes', v_total_step_minutes)
        ),
        (
            'METRIC_SUM',
            'METRIC',
            6,
            v_total_metric_sum,
            'sum of metrics array',
            JSON_OBJECT('sum', v_total_metric_sum)
        ),
        (
            'FACTOR_SUM',
            'METRIC',
            7,
            v_factor_sum,
            'cursor accumulated factor sum',
            JSON_OBJECT('factor_sum', v_factor_sum)
        ),
        (
            'TOP_RUN_ID',
            'METRIC',
            8,
            v_top_run_id,
            'top single weighted run',
            JSON_OBJECT('run_id', v_top_run_id)
        )
    ON DUPLICATE KEY UPDATE
        `group` = VALUES(`group`),
        `rank` = VALUES(`rank`),
        summary_num = VALUES(summary_num),
        summary_text = VALUES(summary_text),
        summary_json = VALUES(summary_json);

    COMMIT;
END//
