    id          BIGINT       NOT NULL AUTO_INCREMENT PRIMARY KEY,
    test_name   VARCHAR(200) NOT NULL,
    detail_text TEXT         NULL,
    created_at  DATETIME(6)  NOT NULL DEFAULT CURRENT_TIMESTAMP(6)
);

CREATE TABLE qb_category (
    category_id        INT          NOT NULL PRIMARY KEY,
    parent_category_id INT          NULL,
    category_name      VARCHAR(100) NOT NULL,
    category_path      VARCHAR(500) NULL,
    is_active          TINYINT(1)   NOT NULL DEFAULT 1,
    created_at         DATETIME(6)  NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    CONSTRAINT fk_qb_category_parent FOREIGN KEY (parent_category_id) REFERENCES qb_category(category_id)
);

CREATE TABLE qb_customer (
    customer_id   INT                                          NOT NULL PRIMARY KEY,
    customer_code VARCHAR(30)                                  NOT NULL UNIQUE,
    customer_name VARCHAR(200)                                 NOT NULL,
    tier          ENUM('BRONZE', 'SILVER', 'GOLD', 'PLATINUM') NOT NULL DEFAULT 'BRONZE',
    status_code   CHAR(1)                                      NOT NULL DEFAULT 'A',
    credit_limit  DECIMAL(18, 2)                               NOT NULL DEFAULT 0,
    meta_json     LONGTEXT                                     NULL,
    note_text     TEXT                                         NULL,
    created_at    DATETIME(6)                                  NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    updated_at    DATETIME(6)                                  NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    CHECK (status_code IN ('A', 'I'))
);

CREATE TABLE qb_product (
    product_id   INT            NOT NULL PRIMARY KEY,
    category_id  INT            NOT NULL,
    sku          VARCHAR(50)    NOT NULL UNIQUE,
    product_name VARCHAR(200)   NOT NULL,
    base_price   DECIMAL(18, 2) NOT NULL,
    tax_rate     DECIMAL(9, 4)  NOT NULL DEFAULT 0.10,
    attr_json    LONGTEXT       NULL,
    is_active    TINYINT(1)     NOT NULL DEFAULT 1,
    created_at   DATETIME(6)    NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    CONSTRAINT fk_qb_product_category FOREIGN KEY (category_id) REFERENCES qb_category(category_id),
    CHECK (base_price >= 0),
    CHECK (tax_rate >= 0)
);

CREATE TABLE qb_inventory (
    product_id     INT         NOT NULL PRIMARY KEY,
    qty_on_hand    INT         NOT NULL,
    reorder_point  INT         NOT NULL,
    warehouse_code VARCHAR(20) NOT NULL,
    updated_at     DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    CONSTRAINT fk_qb_inventory_product FOREIGN KEY (product_id) REFERENCES qb_product(product_id),
    CHECK (qty_on_hand >= 0),
    CHECK (reorder_point >= 0)
);

CREATE TABLE qb_orders (
    order_id     BIGINT                                                NOT NULL PRIMARY KEY,
    customer_id  INT                                                   NOT NULL,
    order_no     VARCHAR(40)                                           NOT NULL UNIQUE,
    order_status ENUM('NEW', 'PAID', 'CANCELLED', 'SHIPPED', 'CLOSED') NOT NULL DEFAULT 'NEW',
    ordered_at   DATETIME(6)                                           NOT NULL,
    paid_at      DATETIME(6)                                           NULL,
    cancelled_at DATETIME(6)                                           NULL,
    memo_text    TEXT                                                  NULL,
    extra_json   LONGTEXT                                              NULL,
    created_at   DATETIME(6)                                           NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    updated_at   DATETIME(6)                                           NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    CONSTRAINT fk_qb_orders_customer FOREIGN KEY (customer_id) REFERENCES qb_customer(customer_id)
);

CREATE TABLE qb_order_item (
    order_item_id BIGINT         NOT NULL PRIMARY KEY,
    order_id      BIGINT         NOT NULL,
    line_no       INT            NOT NULL,
    product_id    INT            NOT NULL,
    qty           INT            NOT NULL,
    unit_price    DECIMAL(18, 2) NOT NULL,
    discount_rate DECIMAL(9, 4)  NOT NULL DEFAULT 0,
    note_text     TEXT           NULL,
    created_at    DATETIME(6)    NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    CONSTRAINT uq_qb_order_item UNIQUE (order_id, line_no),

    CHECK (unit_price >= 0),
    CHECK (discount_rate >= 0 AND discount_rate <= 1)

DELIMITER //

CREATE TRIGGER trg_qb_order_item_ai
    AFTER INSERT
    ON qb_order_item
    FOR EACH ROW
BEGIN
    INSERT INTO qb_audit_log (entity_name, entity_pk, action_name, detail_text)
    VALUES ('qb_order_item', CAST(NEW.order_item_id AS CHAR), 'INSERT', CONCAT('order_id=', NEW.order_id, ',line_no=', NEW.line_no, ',product_id=', NEW.product_id, ',qty=', NEW.qty));
    UPDATE qb_inventory
    SET qty_on_hand = qty_on_hand - NEW.qty
    WHERE product_id = NEW.product_id;
END//

CREATE TRIGGER trg_qb_order_item_au
    AFTER UPDATE
    ON qb_order_item
    FOR EACH ROW
BEGIN
    INSERT INTO qb_audit_log (entity_name, entity_pk, action_name, detail_text)
    VALUES ('qb_order_item', CAST(NEW.order_item_id AS CHAR), 'UPDATE', CONCAT('old_qty=', OLD.qty, ',new_qty=', NEW.qty, ',old_discount=', OLD.discount_rate, ',new_discount=', NEW.discount_rate));
    UPDATE qb_inventory
    SET qty_on_hand = qty_on_hand + OLD.qty - NEW.qty
    WHERE product_id = NEW.product_id;
END//

CREATE TRIGGER trg_qb_order_item_ad
    AFTER DELETE
    ON qb_order_item
    FOR EACH ROW
BEGIN
    INSERT INTO qb_audit_log (entity_name, entity_pk, action_name, detail_text)
    VALUES ('qb_order_item', CAST(OLD.order_item_id AS CHAR), 'DELETE', CONCAT('order_id=', OLD.order_id, ',line_no=', OLD.line_no, ',qty=', OLD.qty));
    UPDATE qb_inventory
    SET qty_on_hand = qty_on_hand + OLD.qty
    WHERE product_id = OLD.product_id;
END//

CREATE PROCEDURE qb_fail(IN p_message TEXT)
BEGIN
    SET @qb_fail_msg = LEFT(COALESCE(p_message, 'UNKNOWN FAILURE'), 128);
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = @qb_fail_msg;
END//

CREATE PROCEDURE qb_assert_true(IN p_condition BOOLEAN, IN p_message TEXT)
BEGIN
    INSERT INTO qb_assert_log (test_name, detail_text)
    VALUES ('ASSERT_TRUE', CONCAT('condition=', IFNULL(CAST(p_condition AS CHAR), 'NULL'), ', message=', p_message));
    IF COALESCE(p_condition, 0) = 0 THEN
        CALL qb_fail(CONCAT('ASSERT_TRUE FAILED: ', p_message));
    END IF;
END//

CREATE PROCEDURE qb_assert_eq_int(IN p_expected BIGINT, IN p_actual BIGINT, IN p_message TEXT)
BEGIN
    INSERT INTO qb_assert_log (test_name, detail_text)
    VALUES ('ASSERT_EQ_INT', CONCAT('expected=', p_expected, ', actual=', p_actual, ', message=', p_message));
    IF COALESCE(p_expected, - 999999999999999) <> COALESCE(p_actual, - 999999999999999) THEN
        CALL qb_fail(CONCAT('ASSERT_EQ_INT FAILED: ', p_message, ' [expected=', p_expected, ', actual=', p_actual, ']'));
    END IF;
END//

CREATE PROCEDURE qb_assert_eq_decimal(
    IN p_expected DECIMAL(30, 8),
    IN p_actual DECIMAL(30, 8),
    IN p_scale INT,
    IN p_message TEXT
)
BEGIN
    DECLARE v_expected DECIMAL(30, 8);
    DECLARE v_actual DECIMAL(30, 8);
    SET v_expected = ROUND(COALESCE(p_expected, 0), p_scale);
    SET v_actual = ROUND(COALESCE(p_actual, 0), p_scale);
    INSERT INTO qb_assert_log (test_name, detail_text)
    VALUES ('ASSERT_EQ_DECIMAL', CONCAT('expected=', v_expected, ', actual=', v_actual, ', scale=', p_scale, ', message=', p_message));
    IF v_expected <> v_actual THEN
        CALL qb_fail(CONCAT('ASSERT_EQ_DECIMAL FAILED: ', p_message, ' [expected=', v_expected, ', actual=', v_actual, ']'));
    END IF;
END//

CREATE FUNCTION qb_fn_order_item_net(
    p_qty INT,
    p_unit_price DECIMAL(18, 2),
    p_discount_rate DECIMAL(9, 4)
)
RETURNS DECIMAL(18, 2)
DETERMINISTIC
BEGIN
    RETURN ROUND(p_qty * p_unit_price * (1 - p_discount_rate), 2);
END//

CREATE FUNCTION qb_fn_customer_grade(
    p_credit_limit DECIMAL(18, 2),
    p_tier VARCHAR(20)
)
RETURNS VARCHAR(20)
DETERMINISTIC
BEGIN
    DECLARE v_grade VARCHAR(20);
    SET v_grade =
        CASE
            WHEN p_tier = 'PLATINUM'
                AND p_credit_limit >= 200000 THEN
                    'S++'
            WHEN p_tier = 'GOLD'
                AND p_credit_limit >= 100000 THEN
                    'S'
            WHEN p_tier = 'SILVER'
                AND p_credit_limit >= 50000 THEN
                    'A'
            WHEN p_tier = 'BRONZE' THEN
                'B'
            ELSE
                'C'
        END;
    RETURN v_grade;
END//

CREATE PROCEDURE qb_rebuild_monthly_summary(
    IN p_from_ym CHAR(7),
    IN p_to_ym CHAR(7)
)
BEGIN
    DECLARE v_sql LONGTEXT;
    DELETE FROM qb_monthly_summary
    WHERE ym BETWEEN p_from_ym AND p_to_ym;
    SET v_sql = '
        INSERT INTO qb_monthly_summary
        (
            ym, customer_id, order_count, gross_amount, discount_amount,
            net_amount, tax_amount, final_amount, running_rank, payload_json
        )
        WITH base AS
        (
            SELECT
                DATE_FORMAT(o.ordered_at, ''%Y-%m'') AS ym,
                o.customer_id,
                o.order_id,
                SUM(oi.qty * oi.unit_price) AS gross_amount,
                SUM(oi.qty * oi.unit_price * oi.discount_rate) AS discount_amount,
                SUM(ROUND(oi.qty * oi.unit_price * (1 - oi.discount_rate), 2)) AS net_amount,
                SUM(ROUND(oi.qty * oi.unit_price * (1 - oi.discount_rate) * p.tax_rate, 2)) AS tax_amount
            FROM qb_orders o
            JOIN qb_order_item oi
              ON oi.order_id = o.order_id
            JOIN qb_product p
              ON p.product_id = oi.product_id
            WHERE DATE_FORMAT(o.ordered_at, ''%Y-%m'') BETWEEN ? AND ?
              AND o.order_status IN (''PAID'', ''SHIPPED'', ''CLOSED'')
            GROUP BY DATE_FORMAT(o.ordered_at, ''%Y-%m''), o.customer_id, o.order_id
        ),
        rolled AS
        (
            SELECT
                ym,
                customer_id,
                COUNT(*) AS order_count,
                ROUND(SUM(gross_amount), 2) AS gross_amount,
                ROUND(SUM(discount_amount), 2) AS discount_amount,
                ROUND(SUM(net_amount), 2) AS net_amount,
                ROUND(SUM(tax_amount), 2) AS tax_amount,
                ROUND(SUM(net_amount + tax_amount), 2) AS final_amount
            FROM base
            GROUP BY ym, customer_id
        )
        SELECT
            r.ym,
            r.customer_id,
            r.order_count,
            r.gross_amount,
            r.discount_amount,
            r.net_amount,
            r.tax_amount,
            r.final_amount,
            DENSE_RANK() OVER (PARTITION BY r.ym ORDER BY r.final_amount DESC, r.customer_id) AS running_rank,
            JSON_OBJECT(
                ''customer_id'', r.customer_id,
                ''ym'', r.ym,
                ''order_count'', r.order_count,
                ''gross_amount'', r.gross_amount,
                ''discount_amount'', r.discount_amount,
                ''net_amount'', r.net_amount,
                ''tax_amount'', r.tax_amount,
                ''final_amount'', r.final_amount
            ) AS payload_json
        FROM rolled r
    ';
    PREPARE stmt
    FROM v_sql;
    EXECUTE stmt
    USING p_from_ym,
        p_to_ym;
    DEALLOCATE PREPARE stmt;
END//

CREATE PROCEDURE qb_cursor_inventory_scan(OUT p_low_stock_count INT)
BEGIN
    DECLARE v_done INT DEFAULT 0;
    DECLARE v_product_id INT;
    DECLARE v_qty INT;
    DECLARE v_reorder INT;
    DECLARE cur CURSOR FOR
        SELECT product_id,
            qty_on_hand,
            reorder_point
        FROM qb_inventory
        ORDER BY product_id;
    DECLARE CONTINUE HANDLER FOR NOT FOUND
        SET v_done = 1;
    SET p_low_stock_count = 0;
    OPEN cur;
    read_loop: LOOP
        FETCH cur
        INTO v_product_id,
            v_qty,
            v_reorder;
        IF v_done = 1 THEN
            LEAVE read_loop;
        END IF;
        IF v_qty <= v_reorder THEN
            SET p_low_stock_count = p_low_stock_count + 1;
        END IF;
    END LOOP;
    CLOSE cur;
END//

CREATE PROCEDURE qb_make_test_order(
    IN p_order_id BIGINT,
    IN p_customer_id INT,
    IN p_order_no VARCHAR(40),
    IN p_ordered_at DATETIME(6),
    IN p_product_id INT,
    IN p_qty INT,
    IN p_unit_price DECIMAL(18, 2),
    IN p_discount_rate DECIMAL(9, 4)
)
BEGIN
    DECLARE v_exists INT DEFAULT 0;
    SELECT COUNT(*)
    INTO v_exists
    FROM qb_customer
    WHERE customer_id = p_customer_id
        AND status_code = 'A';
    IF v_exists = 0 THEN
        CALL qb_fail(CONCAT('customer invalid or inactive: ', p_customer_id));
    END IF;
    START TRANSACTION;
    INSERT INTO qb_orders (order_id, customer_id, order_no, order_status, ordered_at, paid_at, cancelled_at, memo_text, extra_json)
    VALUES (p_order_id, p_customer_id, p_order_no, 'PAID', p_ordered_at, p_ordered_at, NULL, 'generated by qb_make_test_order; contains ; and ''quotes''', JSON_OBJECT('source', 'proc', 'product_id', p_product_id));
    INSERT INTO qb_order_item (order_item_id, order_id, line_no, product_id, qty, unit_price, discount_rate, note_text)
    VALUES (p_order_id * 10, p_order_id, 1, p_product_id, p_qty, p_unit_price, p_discount_rate, CONCAT('auto line for order=', p_order_id));
    COMMIT;
END//

CREATE OR REPLACE VIEW qb_v_customer_rollup AS
    SELECT c.customer_id,
        c.customer_code,
        c.customer_name,
        c.tier,
        qb_fn_customer_grade(c.credit_limit, c.tier) AS customer_grade,
        COALESCE(o.order_count, 0) AS order_count,
        COALESCE(o.total_net_amount, 0.00) AS total_net_amount,
        COALESCE(o.avg_order_amount, 0.00) AS avg_order_amount,
        COALESCE(j.tag_count, 0) AS tag_count,
    CASE
        WHEN COALESCE(o.total_net_amount, 0) >= 3000 THEN 'VIP'
        WHEN COALESCE(o.total_net_amount, 0) >= 1000 THEN 'CORE'
        WHEN COALESCE(o.total_net_amount, 0) > 0 THEN 'ACTIVE'
        ELSE 'NO_PAID_ORDER'
    END AS spending_band
    FROM qb_customer c
    LEFT JOIN (
        SELECT o.customer_id,
            COUNT(DISTINCT o.order_id) AS order_count,
            ROUND(SUM(qb_fn_order_item_net(oi.qty, oi.unit_price, oi.discount_rate)), 2) AS total_net_amount,
            ROUND(SUM(qb_fn_order_item_net(oi.qty, oi.unit_price, oi.discount_rate)) / COUNT(DISTINCT o.order_id), 2) AS avg_order_amount
        FROM qb_orders o
        JOIN qb_order_item oi
            ON oi.order_id = o.order_id
        WHERE o.order_status IN ('PAID', 'SHIPPED', 'CLOSED')
        GROUP BY o.customer_id
    ) o
    ON o.customer_id = c.customer_id
    LEFT JOIN (
        SELECT c2.customer_id,
            JSON_LENGTH(JSON_EXTRACT(c2.meta_json, '$.tags')) AS tag_count
        FROM qb_customer c2
    ) j
    ON j.customer_id = c.customer_id;

WITH paid_order_line AS (
    SELECT
        o.customer_id,
        o.order_id,
        oi.order_item_id,
        oi.product_id,
        oi.qty,
        oi.unit_price,
        oi.discount_rate,
        qb_fn_order_item_net(oi.qty, oi.unit_price, oi.discount_rate) AS net_amount
    FROM qb_orders o
    JOIN qb_order_item oi
        ON oi.order_id = o.order_id
    WHERE o.order_status IN ('PAID', 'SHIPPED', 'CLOSED')
),
ranked AS (
    SELECT
        pol.*,
        ROW_NUMBER() OVER (
            PARTITION BY pol.customer_id
            ORDER BY pol.net_amount DESC,
            pol.order_item_id
        ) AS rn
    FROM paid_order_line pol
)
SELECT COUNT(*)
INTO @top_line_count
FROM ranked
WHERE rn = 1;

CALL qb_assert_eq_int(3, @top_line_count, 'top line per active purchasing customer');
