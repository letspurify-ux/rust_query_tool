CREATE TABLE node (
    node_id INT NOT NULL,
    parent_node_id INT NULL,
    `group` VARCHAR(20) NOT NULL,
    node_name VARCHAR(100) NOT NULL,
    sort_no INT NOT NULL,
    meta JSON NOT NULL,
    path_hint VARCHAR(200)
        GENERATED ALWAYS AS (JSON_UNQUOTE(JSON_EXTRACT(meta, '$.pathHint'))) VIRTUAL,
    PRIMARY KEY (node_id),
    UNIQUE KEY uk_node_group_name (`group`, node_name),
    KEY idx_node_parent (parent_node_id),
    CONSTRAINT fk_node_parent
        FOREIGN KEY (parent_node_id) REFERENCES node (node_id),
    CONSTRAINT chk_node_self
        CHECK (parent_node_id IS NULL OR parent_node_id <> node_id)
) ENGINE = InnoDB;

CREATE TABLE task (
    task_id BIGINT NOT NULL,
    node_id INT NOT NULL,
    owner_name VARCHAR(100) NOT NULL,
    status_code VARCHAR(20) NOT NULL,
    priority INT NOT NULL,
    `rank` INT NOT NULL,
    payload JSON NOT NULL,
    created_at DATETIME(6) NOT NULL,
    first_tag VARCHAR(50)
        GENERATED ALWAYS AS (JSON_UNQUOTE(JSON_EXTRACT(payload, '$.tags[0]'))) STORED,
    tag_count INT
        GENERATED ALWAYS AS (JSON_LENGTH(JSON_EXTRACT(payload, '$.tags'))) STORED,
    note_text VARCHAR(255)
        GENERATED ALWAYS AS (JSON_UNQUOTE(JSON_EXTRACT(payload, '$.note'))) VIRTUAL,
    PRIMARY KEY (task_id),
    UNIQUE KEY uk_task_owner_rank (owner_name, `rank`),
    KEY idx_task_node_status (node_id, status_code),
    KEY idx_task_owner_priority (owner_name, priority DESC),
    CONSTRAINT fk_task_node
        FOREIGN KEY (node_id) REFERENCES node (node_id),
    CONSTRAINT chk_task_priority
        CHECK (priority BETWEEN 1 AND 9)
) ENGINE = InnoDB;

CREATE TABLE task_log (
    log_id BIGINT NOT NULL AUTO_INCREMENT,
    task_id BIGINT NOT NULL,
    log_type VARCHAR(30) NOT NULL,
    old_status VARCHAR(20) NULL,
    new_status VARCHAR(20) NULL,
    log_note VARCHAR(255) NOT NULL,
    created_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    PRIMARY KEY (log_id),
    KEY idx_task_log_task (task_id)
) ENGINE = InnoDB;

CREATE TABLE agg_result (
    result_key VARCHAR(100) NOT NULL,
    `group` VARCHAR(20) NOT NULL,
    result_num DECIMAL(18, 4) NULL,
    result_text VARCHAR(500) NULL,
    result_json JSON NULL,
    created_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    PRIMARY KEY (result_key)
) ENGINE = InnoDB;

CREATE TABLE error_log (
    log_id BIGINT NOT NULL AUTO_INCREMENT,
    step_name VARCHAR(100) NOT NULL,
    sql_state CHAR(5) NULL,
    mysql_errno INT NULL,
    message_text VARCHAR(512) NULL,
    created_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
    PRIMARY KEY (log_id)
) ENGINE = InnoDB;

CREATE FUNCTION fn_priority_weight (p_priority INT)
RETURNS DECIMAL(10, 2)
DETERMINISTIC
BEGIN
    RETURN CASE
        WHEN p_priority >= 9 THEN 3.50
        WHEN p_priority >= 7 THEN 2.25
        WHEN p_priority >= 5 THEN 1.50
        ELSE 1.00
    END;
END$$

CREATE PROCEDURE sp_check (
    IN p_condition BOOLEAN,
    IN p_message VARCHAR(255)
)
BEGIN
    IF NOT p_condition THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = p_message;
    END IF;
END$$

CREATE PROCEDURE sp_touch_rank (
    INOUT p_rank INT,
    IN p_delta INT
)
BEGIN
    SET p_rank = p_rank + p_delta;
END$$

CREATE PROCEDURE sp_status_counts (
    OUT p_done_cnt INT,
    OUT p_running_cnt INT
)
BEGIN
    SELECT
        SUM(CASE WHEN status_code = 'DONE' THEN 1 ELSE 0 END),
        SUM(CASE WHEN status_code = 'RUNNING' THEN 1 ELSE 0 END)
    INTO
        p_done_cnt,
        p_running_cnt
    FROM task;
END$$

CREATE TRIGGER trg_task_bi
BEFORE INSERT ON task
FOR EACH ROW
BEGIN
    SET NEW.owner_name = LOWER(TRIM(NEW.owner_name));
    SET NEW.status_code = UPPER(TRIM(NEW.status_code));

    IF NEW.priority NOT BETWEEN 1 AND 9 THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'priority must be between 1 and 9';
    END IF;
END$$

CREATE TRIGGER trg_task_bu
BEFORE UPDATE ON task
FOR EACH ROW
BEGIN
    SET NEW.owner_name = LOWER(TRIM(NEW.owner_name));
    SET NEW.status_code = UPPER(TRIM(NEW.status_code));
END$$

CREATE TRIGGER trg_task_ai
AFTER INSERT ON task
FOR EACH ROW
BEGIN
    INSERT INTO task_log (
        task_id,
        log_type,
        old_status,
        new_status,
        log_note
    )
    VALUES (
        NEW.task_id,
        'INSERT',
        NULL,
        NEW.status_code,
        CONCAT(
            'inserted first_tag=',
            COALESCE(JSON_UNQUOTE(JSON_EXTRACT(NEW.payload, '$.tags[0]')), 'null')
        )
    );
END$$

CREATE TRIGGER trg_task_au
AFTER UPDATE ON task
FOR EACH ROW
BEGIN
    IF NOT (OLD.status_code <=> NEW.status_code) THEN
        INSERT INTO task_log (
            task_id,
            log_type,
            old_status,
            new_status,
            log_note
        )
        VALUES (
            NEW.task_id,
            'STATUS_CHANGE',
            OLD.status_code,
            NEW.status_code,
            CONCAT(OLD.status_code, ' -> ', NEW.status_code)
        );
    END IF;
END$$
