    dept_name      VARCHAR2(100) NOT NULL,
    parent_dept_id NUMBER,
    created_at     DATE          DEFAULT SYSDATE NOT NULL,
    status         CHAR(1)       DEFAULT 'A' CHECK(status IN('A', 'I')),

CREATE TABLE qt_emp (
    emp_id         NUMBER             GENERATED BY DEFAULT AS IDENTITY CONSTRAINT qt_emp_pk PRIMARY KEY,
    dept_id        NUMBER             NOT NULL CONSTRAINT qt_emp_fk_dept REFERENCES qt_dept(dept_id),

    status                       VARCHAR2(20)  DEFAULT 'NEW' CHECK(status IN('NEW', 'PAID', 'CANCEL')),
    
remark      VARCHAR2(4000)

);

CREATE OR REPLACE VIEW qt_emp_sales_v AS
WITH sale_base AS (
    SELECT
        e.emp_id,
        e.emp_name,
        d.dept_name,
        s.sale_dt,
        s.amount,
        s.tax,
        s.status,
        TRUNC (s.sale_dt, 'MM') AS sale_month,
        SUM (s.amount) OVER (PARTITION BY e.emp_id) AS total_amount_by_emp,
        ROW_NUMBER () OVER (PARTITION BY e.emp_id ORDER BY s.sale_dt DESC, s.amount DESC) AS rn
    FROM qt_emp e
    JOIN qt_dept d
        ON d.dept_id = e.dept_id
    LEFT JOIN qt_sales s
        ON s.emp_id = e.emp_id
)
SELECT *
FROM sale_base;

CREATE OR REPLACE TYPE qt_obj_emp AS
    OBJECT (emp_id NUMBER, emp_name VARCHAR2 (100), dept_name VARCHAR2 (100), total_sales NUMBER);
/

CREATE OR REPLACE TYPE qt_obj_emp_tab AS
    TABLE OF qt_obj_emp;
/

CREATE OR REPLACE FUNCTION qt_pipe_emp (p_dept_id NUMBER DEFAULT NULL) RETURN qt_obj_emp_tab PIPELINED IS
BEGIN
    FOR r IN (
        SELECT e.emp_id,
            e.emp_name,
            d.dept_name,
            NVL (SUM (s.amount), 0) AS total_sales
        FROM qt_emp e
        JOIN qt_dept d
            ON d.dept_id = e.dept_id
        LEFT JOIN qt_sales s
            ON s.emp_id = e.emp_id
        WHERE p_dept_id IS NULL
            OR e.dept_id = p_dept_id
        GROUP BY e.emp_id,
            e.emp_name,
            d.dept_name
        ORDER BY e.emp_id
    ) LOOP
        PIPE ROW (qt_obj_emp (r.emp_id, r.emp_name, r.dept_name, r.total_sales));
    END LOOP;
    RETURN;
END qt_pipe_emp;
/

CREATE OR REPLACE TRIGGER qt_sales_ct
    FOR INSERT OR UPDATE OR DELETE ON qt_sales COMPOUND TRIGGER TYPE t_num_tab IS
    TABLE OF NUMBER INDEX BY PLS_INTEGER;
    g_emp_ids t_num_tab;
    g_idx PLS_INTEGER := 0;
    BEFORE STATEMENT IS
    BEGIN
        g_idx := 0;
    END BEFORE STATEMENT;

    AFTER EACH ROW IS
    BEGIN
        g_idx := g_idx + 1;
        IF INSERTING
            OR UPDATING THEN
            g_emp_ids (g_idx) := :NEW.emp_id;
        ELSE
            g_emp_ids (g_idx) := :OLD.emp_id;
        END IF;
    END AFTER EACH ROW;

    AFTER STATEMENT IS
    BEGIN
        IF g_idx > 0 THEN
            INSERT INTO qt_audit_log (module_name, action_name, log_text)
            VALUES ('qt_sales_ct', 'AFTER STATEMENT', 'Rows affected=' || g_idx || ', sample emp_id=' || g_emp_ids (1));
        END IF;
    END AFTER STATEMENT;

END qt_sales_ct;

CREATE OR REPLACE PACKAGE BODY qt_pkg_extreme IS
    g_run_mode VARCHAR2 (30);
    PROCEDURE log_msg (p_module VARCHAR2, p_action VARCHAR2, p_text CLOB) IS
        PRAGMA AUTONOMOUS_TRANSACTION;
    BEGIN
        INSERT INTO qt_audit_log (module_name, action_name, log_text)
        VALUES (p_module, p_action, p_text);
        COMMIT;
    EXCEPTION
        WHEN OTHERS THEN
            ROLLBACK;
            RAISE;
    END log_msg;

    FUNCTION normalize_name (p_name VARCHAR2) RETURN VARCHAR2 DETERMINISTIC IS
    BEGIN
        RETURN REGEXP_REPLACE (UPPER (TRIM (p_name)), '\s+', ' ');
    END normalize_name;

    FUNCTION json_like_report (p_emp_id NUMBER) RETURN CLOB IS
        v_report CLOB;
        v_name qt_emp.emp_name%TYPE;
        v_dept qt_dept.dept_name%TYPE;
        v_total NUMBER;
    BEGIN
        SELECT e.emp_name,
            d.dept_name,
            NVL (SUM (s.amount), 0)
        INTO v_name,
            v_dept,
            v_total
        FROM qt_emp e
        JOIN qt_dept d
            ON d.dept_id = e.dept_id
        LEFT JOIN qt_sales s
            ON s.emp_id = e.emp_id
        WHERE e.emp_id = p_emp_id
        GROUP BY e.emp_name,
            d.dept_name;
        v_report := '{' || '"empId":' || p_emp_id || ',"empName":"' || REPLACE (v_name, '"', '\"') || '"' || ',"deptName":"' || REPLACE (v_dept, '"', '\"') || '"' || ',"totalSales":' || TO_CHAR (v_total, 'FM9999990D00', 'NLS_NUMERIC_CHARACTERS=.,') || '}';
        RETURN v_report;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RETURN '{"empId":null,"error":"NO_DATA_FOUND"}';
    END json_like_report;

    PROCEDURE load_bonus_sales (p_year NUMBER, p_debug BOOLEAN DEFAULT FALSE) IS
        TYPE t_sales_src_rec IS RECORD (emp_id qt_emp.emp_id%TYPE, sale_dt DATE, amount NUMBER (12, 2), tax NUMBER (12, 2), 
        remark   VARCHAR2(4000)
        );
        TYPE t_sales_src_tab IS TABLE OF t_sales_src_rec;
        l_rows t_sales_src_tab;
        l_failures PLS_INTEGER := 0;
        l_inserted PLS_INTEGER := 0;
        CURSOR c_src IS
        SELECT e.emp_id,
            ADD_MONTHS (TO_DATE (TO_CHAR (p_year) || '-01-01', 'YYYY-MM-DD'), MOD (e.emp_id, 12)) AS sale_dt,
            ROUND (NVL (e.salary, 0) * 0.07, 2) AS amount,
            ROUND (NVL (e.salary, 0) * 0.007, 2) AS tax,
            q'[bonus-generated ; / / * not a comment * / -- literal]' AS REMARK
        FROM qt_emp e
        WHERE EXTRACT (YEAR
        FROM e.hired_at) <= p_year
            AND e.active_yn = 'Y';
        PROCEDURE DEBUG (p_text VARCHAR2) IS
        BEGIN
            IF p_debug THEN
                DBMS_OUTPUT.PUT_LINE ('[DEBUG] ' || p_text);
            END IF;
        END DEBUG;
    BEGIN
        OPEN c_src;
        FETCH c_src BULK COLLECT
        INTO l_rows;
        CLOSE c_src;
        DEBUG ('Rows to insert=' || l_rows.COUNT);
        FORALL i IN INDICES OF l_rows SAVE EXCEPTIONS
        INSERT INTO qt_sales (sale_id, emp_id, sale_dt, amount, tax, status, 
        remark
        )
        VALUES (NULL, l_rows (i).emp_id, l_rows (i).sale_dt, l_rows (i).amount, l_rows (i).tax, 'NEW', l_rows (i).REMARK);
        l_inserted := SQL%ROWCOUNT;
        log_msg ('qt_pkg_extreme', 'load_bonus_sales', 'Inserted rows=' || l_inserted);
    EXCEPTION
        WHEN OTHERS THEN
            IF SQLCODE = - 24381 THEN
                l_failures := SQL%BULK_EXCEPTIONS.COUNT;
                FOR i IN 1..l_failures LOOP
                    log_msg ('qt_pkg_extreme', 'bulk_error', 'idx=' || SQL%BULK_EXCEPTIONS (i).ERROR_INDEX || ', code=' || SQL%BULK_EXCEPTIONS (i).ERROR_CODE);
                END LOOP;
            ELSE
                log_msg ('qt_pkg_extreme', 'load_bonus_sales_error', SQLERRM);
            END IF;
            RAISE;
    END load_bonus_sales;

    PROCEDURE open_emp_cursor (p_dept_id IN NUMBER, p_min_amount IN NUMBER DEFAULT 0, p_rc OUT SYS_REFCURSOR) IS
        v_sql CLOB;
    BEGIN
        v_sql := q'[
            WITH sales_agg AS (
                SELECT
                    e.emp_id,
                    e.emp_name,
                    d.dept_name,
                    NVL(SUM(s.amount), 0) AS total_amount,
                    COUNT(s.sale_id) AS sale_count,
                    MAX(s.sale_dt) AS last_sale_dt
                FROM qt_emp e
                JOIN qt_dept d
                  ON d.dept_id = e.dept_id
                LEFT JOIN qt_sales s
                  ON s.emp_id = e.emp_id
                WHERE e.dept_id = :b1
                GROUP BY e.emp_id, e.emp_name, d.dept_name
            )
            SELECT
                emp_id,
                emp_name,
                dept_name,
                total_amount,
                sale_count,
                last_sale_dt,
                CASE
                    WHEN total_amount >= 10000 THEN 'VIP'
                    WHEN total_amount >= 3000 THEN 'CORE'
                    ELSE 'EDGE'
                END AS grade
            FROM sales_agg
            WHERE total_amount >= :b2
            ORDER BY total_amount DESC, emp_name
        ]';
        OPEN p_rc FOR
            v_sql USING p_dept_id,
                p_min_amount;
    END open_emp_cursor;
BEGIN
    g_run_mode :=
    CASE
        WHEN TO_CHAR (SYSDATE, 'DY', 'NLS_DATE_LANGUAGE=ENGLISH') IN ('SAT', 'SUN') THEN
            'WEEKEND'
        ELSE
            'WEEKDAY'
    END;
    log_msg ('qt_pkg_extreme', 'init', 'package initialized, mode=' || g_run_mode);
END qt_pkg_extreme;

MERGE INTO qt_sales dst USING (
    WITH bonus_src AS (
        SELECT
            e.emp_id,
            TRUNC (SYSDATE, 'MM') AS sale_dt,
            ROUND (NVL (e.salary, 0) *
                CASE
                    WHEN e.dept_id = 20 THEN 0.15
                    ELSE 0.05
                END, 2) AS amount,
            ROUND (NVL (e.salary, 0) * 0.01, 2) AS tax,
            'MERGE-UPSERT' AS REMARK
        FROM qt_emp e
        WHERE e.active_yn = 'Y'
    )
    SELECT *
    FROM bonus_src
) src
    ON (dst.emp_id = src.emp_id
        AND dst.sale_dt = src.sale_dt
        AND dst.REMARK = src.REMARK)
    WHEN MATCHED THEN
UPDATE
SET dst.amount = src.amount,
    dst.tax = src.tax,
    dst.status = 'PAID'
DELETE
WHERE dst.amount = 0
    WHEN NOT MATCHED THEN
INSERT (sale_id, emp_id, sale_dt, amount, tax, status, 
remark
)
VALUES (NULL, src.emp_id, src.sale_dt, src.amount, src.tax, 'NEW', src.REMARK);

UPDATE qt_emp e
SET e.salary = e.salary + (
    SELECT NVL (ROUND (AVG (s.amount) / 100, 2), 0)
    FROM qt_sales s
    WHERE s.emp_id = e.emp_id
),
    e.note = NVL (e.note, EMPTY_CLOB ()) || CHR (10) || 'UPDATED@' || TO_CHAR (SYSTIMESTAMP, 'YYYY-MM-DD HH24:MI:SS.FF3')
WHERE e.active_yn = 'Y'
    AND EXISTS (
        SELECT 1
        FROM qt_sales s
        WHERE s.emp_id = e.emp_id
            AND s.status IN ('NEW', 'PAID')
    );

DELETE
FROM qt_sales s
WHERE s.status = 'CANCEL'
    AND EXISTS (
        SELECT 1
        FROM qt_emp e
        WHERE e.emp_id = s.emp_id
            AND e.active_yn = 'N'
    );

WITH dept_tree (dept_id, dept_name, parent_dept_id, lvl, path_txt) AS (
    SELECT
        d.dept_id,
        d.dept_name,
        d.parent_dept_id,
        1 AS lvl,
        TO_CHAR (d.dept_id) AS path_txt
    FROM qt_dept d
    WHERE d.parent_dept_id IS NULL
    UNION ALL
    SELECT
        c.dept_id,
        c.dept_name,
        c.parent_dept_id,
        p.lvl + 1 AS lvl,
        p.path_txt || '>' || c.dept_id AS path_txt
    FROM qt_dept c
    JOIN dept_tree p
        ON p.dept_id = c.parent_dept_id
),
sales_ranked AS (
    SELECT /*+ MATERIALIZE */
        e.emp_id,
        e.emp_name,
        d.dept_name,
        s.sale_dt,
        s.amount,
        SUM (s.amount) OVER (PARTITION BY e.emp_id) AS total_by_emp,
        DENSE_RANK () OVER (PARTITION BY d.dept_name ORDER BY s.amount DESC NULLS LAST) AS dept_rank,
        LAG (s.amount) OVER (PARTITION BY e.emp_id ORDER BY s.sale_dt) AS prev_amount,
        LEAD (s.amount) OVER (PARTITION BY e.emp_id ORDER BY s.sale_dt) AS next_amount
    FROM qt_emp e
    JOIN qt_dept d
        ON d.dept_id = e.dept_id
    LEFT JOIN qt_sales s
        ON s.emp_id = e.emp_id
)
SELECT
    t.lvl,
    t.path_txt,
    r.dept_name,
    r.emp_id,
    r.emp_name,
    r.sale_dt,
    r.amount,
    r.total_by_emp,
    r.dept_rank,
    NVL (r.amount - r.prev_amount, 0) AS diff_from_prev,
    NVL (r.next_amount - r.amount, 0) AS diff_to_next
FROM dept_tree t
JOIN sales_ranked r
    ON r.dept_name = t.dept_name
WHERE EXISTS (
        SELECT 1
        FROM qt_emp e2
        WHERE e2.emp_id = r.emp_id
            AND e2.active_yn = 'Y'
    )
ORDER BY t.path_txt,
    r.emp_name,
    r.sale_dt;

------------------------------------------------------------------------
-- 13. PIVOT
------------------------------------------------------------------------

SELECT *
FROM (
        SELECT TO_CHAR (sale_dt, 'YYYY-MM') AS sale_month,
            status,
            amount
        FROM qt_sales
    ) PIVOT (SUM (amount)
FOR status IN ('NEW' AS new_amt, 'PAID' AS paid_amt, 'CANCEL' AS cancel_amt))
ORDER BY sale_month;

------------------------------------------------------------------------
-- 14. UNPIVOT
------------------------------------------------------------------------

SELECT *
FROM (
        SELECT emp_id,
            salary,
            NVL (commission_pct, 0) AS commission_pct
        FROM qt_emp
    ) UNPIVOT INCLUDE NULLS (metric_value
FOR metric_name IN (salary AS 'SALARY', commission_pct AS 'COMMISSION'))
ORDER BY emp_id,
    metric_name;

------------------------------------------------------------------------
-- 15. CROSS APPLY
------------------------------------------------------------------------

SELECT e.emp_id,
    e.emp_name,
    x.max_amt,
    x.min_amt
FROM qt_emp e CROSS APPLY (
        SELECT MAX (s.amount) AS max_amt,
            MIN (s.amount) AS min_amt
        FROM qt_sales s
        WHERE s.emp_id = e.emp_id
    ) x
ORDER BY e.emp_id;

------------------------------------------------------------------------
-- 16. GROUPING SETS + LISTAGG
------------------------------------------------------------------------

SELECT NVL (d.dept_name, 'ALL') AS dept_name,
    NVL (e.active_yn, '*') AS active_yn,
    SUM (e.salary) AS total_salary
FROM qt_emp e
JOIN qt_dept d
    ON d.dept_id = e.dept_id
GROUP BY GROUPING SETS ((d.dept_name, e.active_yn), (d.dept_name), ())
ORDER BY dept_name,
    active_yn;

SELECT d.dept_name,
    LISTAGG (e.emp_name, ', ') WITHIN
    GROUP (ORDER BY e.emp_name) AS emp_list
FROM qt_dept d
LEFT JOIN qt_emp e
    ON e.dept_id = d.dept_id
GROUP BY d.dept_name
ORDER BY d.dept_name;

-- 17. CONNECT BY
------------------------------------------------------------------------

SELECT LEVEL AS lvl,
    LPAD (' ', (LEVEL - 1) * 2) || dept_name AS tree_name,
    SYS_CONNECT_BY_PATH (dept_name, ' > ') AS path_txt
FROM qt_dept
START WITH parent_dept_id IS NULL
CONNECT BY PRIOR dept_id = parent_dept_id
ORDER SIBLINGS BY dept_name;

DECLARE
    TYPE t_varchar_tab IS TABLE OF VARCHAR2 (100);
    l_names t_varchar_tab := t_varchar_tab ('  Alice  ', 'Bob', q'[Carol / ; --]', 'D''Angelo');
    l_rc SYS_REFCURSOR;
    l_emp_id NUMBER;
    l_emp_name VARCHAR2 (100);
    l_dept_name VARCHAR2 (100);
    l_total_amount NUMBER;
    l_sale_count NUMBER;
    l_last_sale_dt DATE;
    l_grade VARCHAR2 (20);
    l_sale_rows NUMBER;
    l_sale_sum NUMBER;
    PROCEDURE print_line (p_txt VARCHAR2) IS
    BEGIN
        DBMS_OUTPUT.PUT_LINE (p_txt);
    END print_line;

    FUNCTION safe_div (p_num NUMBER, p_den NUMBER) RETURN NUMBER IS
    BEGIN
        RETURN
        CASE
            WHEN p_den = 0 THEN
                NULL
            ELSE
                p_num / p_den
        END;
    END safe_div;

BEGIN
    SAVEPOINT before_bonus;
    qt_pkg_extreme.load_bonus_sales (p_year => 2025, p_debug => TRUE);
    qt_pkg_extreme.open_emp_cursor (p_dept_id => 20, p_min_amount => 0, p_rc => l_rc);
    LOOP
        FETCH l_rc
        INTO l_emp_id,
            l_emp_name,
            l_dept_name,
            l_total_amount,
            l_sale_count,
            l_last_sale_dt,
            l_grade;
        EXIT WHEN l_rc%NOTFOUND;
        print_line ('CURSOR => emp_id=' || l_emp_id || ', name=' || l_emp_name || ', grade=' || l_grade || ', avg=' || TO_CHAR (safe_div (l_total_amount, NULLIF (l_sale_count, 0)), 'FM9999990D00', 'NLS_NUMERIC_CHARACTERS=.,'));
    END LOOP;
    CLOSE l_rc;
    <<outer_loop>>
    FOR i IN 1..l_names.COUNT LOOP
        l_names (i) := qt_pkg_extreme.normalize_name (l_names (i));
        CONTINUE WHEN l_names (i) IS NULL;
        print_line ('normalized[' || i || ']=' || l_names (i));
        FOR j IN REVERSE 1..3 LOOP
            EXIT outer_loop WHEN i = 4
                AND j = 2;
            NULL;
        END LOOP;
    END LOOP;
    SELECT COUNT (*),
        NVL (SUM (amount), 0)
    INTO l_sale_rows,
        l_sale_sum
    FROM qt_sales
    WHERE status IN ('NEW', 'PAID');
    CASE
        WHEN l_sale_sum >= 20000 THEN
            print_line ('CASE => HIGH');
        WHEN l_sale_sum > 0 THEN
            print_line ('CASE => NORMAL');
        ELSE
            print_line ('CASE => EMPTY');
    END CASE;
    IF l_rc%ISOPEN THEN
        CLOSE l_rc;
    END IF;
EXCEPTION
    WHEN OTHERS THEN
        IF l_rc%ISOPEN THEN
            CLOSE l_rc;
        END IF;
        ROLLBACK TO before_bonus;
        RAISE;
END;

COMMIT;
