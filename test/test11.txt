CREATE TABLE qt_departments (
    dept_id        NUMBER        CONSTRAINT qt_departments_pk PRIMARY KEY,
    dept_name      VARCHAR2(100) CONSTRAINT qt_departments_uq UNIQUE NOT NULL,
    parent_dept_id NUMBER        NULL,
    status         VARCHAR2(10)  DEFAULT 'ACTIVE' NOT NULL,
    created_at     TIMESTAMP     DEFAULT SYSTIMESTAMP NOT NULL,
    CONSTRAINT qt_departments_fk_parent FOREIGN KEY(parent_dept_id) REFERENCES qt_departments(dept_id),
    CONSTRAINT qt_departments_status_ck CHECK(status IN('ACTIVE', 'INACTIVE'))
);
/

CREATE TABLE qt_employees (
    emp_id       NUMBER             CONSTRAINT qt_employees_pk PRIMARY KEY,
    dept_id      NUMBER             NOT NULL,
    manager_id   NUMBER             NULL,
    emp_name     VARCHAR2(200 CHAR) NOT NULL,
    email        VARCHAR2(320),
    hire_date    DATE               DEFAULT TRUNC(SYSDATE) NOT NULL,
    salary       NUMBER(12, 2),
    bonus_pct    NUMBER(5, 2),
    profile_json CLOB,
    remarks      VARCHAR2(4000),
    upper_name     VARCHAR2(320),
    CONSTRAINT qt_employees_fk_dept FOREIGN KEY(dept_id) REFERENCES qt_departments(dept_id),
    CONSTRAINT qt_employees_fk_manager FOREIGN KEY(manager_id) REFERENCES qt_employees(emp_id),
    CONSTRAINT qt_employees_profile_json_ck CHECK(profile_json IS JSON)
);
/

CREATE TABLE qt_sales (
    sale_id    NUMBER         CONSTRAINT qt_sales_pk PRIMARY KEY,
    emp_id     NUMBER         NOT NULL,
    sale_date  DATE           NOT NULL,
    amount     NUMBER(12, 2)  NOT NULL,
    status     VARCHAR2(20)   NOT NULL,
    note       VARCHAR2(1000),
    created_at TIMESTAMP      DEFAULT SYSTIMESTAMP NOT NULL,
    CONSTRAINT qt_sales_fk_emp FOREIGN KEY(emp_id) REFERENCES qt_employees(emp_id),
    CONSTRAINT qt_sales_status_ck CHECK(status IN('NEW', 'PAID', 'CANCELLED', 'PENDING'))
);
/

CREATE TABLE qt_audit_log (
    log_id      NUMBER        GENERATED ALWAYS AS IDENTITY CONSTRAINT qt_audit_log_pk PRIMARY KEY,
    log_ts      TIMESTAMP     DEFAULT SYSTIMESTAMP NOT NULL,

CREATE OR REPLACE PACKAGE qt_torture_pkg AUTHID DEFINER IS
    SUBTYPE t_name IS VARCHAR2 (200 CHAR);
    TYPE t_refcur IS REF CURSOR;
    TYPE t_num_nt IS TABLE OF NUMBER;
    TYPE t_name_aat IS TABLE OF t_name INDEX BY PLS_INTEGER;
    g_run_id NUMBER := 0;
    PROCEDURE AUDIT (p_module VARCHAR2, p_action VARCHAR2, p_detail CLOB);
    FUNCTION get_employee_report (p_dept_id NUMBER, p_min_salary NUMBER DEFAULT 0) RETURN t_refcur;
    FUNCTION pipe_sales (p_emp_id NUMBER DEFAULT NULL) RETURN qt_sale_obj_tab PIPELINED;
    PROCEDURE complex_block (p_dept_id NUMBER, p_raise_pct NUMBER DEFAULT 1.5, p_mode VARCHAR2 DEFAULT 'DEEP');
END qt_torture_pkg;
/

/* ---------------------------------------------------------------------
   05. PACKAGE BODY
   --------------------------------------------------------------------- */

CREATE OR REPLACE PACKAGE BODY qt_torture_pkg IS
    PROCEDURE AUDIT (p_module VARCHAR2, p_action VARCHAR2, p_detail CLOB) IS
        PRAGMA AUTONOMOUS_TRANSACTION;
    BEGIN
        INSERT INTO qt_audit_log (module_name, action_name, detail_text)
        VALUES (p_module, p_action, p_detail);
        COMMIT;
    EXCEPTION
        WHEN OTHERS THEN
            ROLLBACK;
    END AUDIT;

    FUNCTION get_employee_report (p_dept_id NUMBER, p_min_salary NUMBER DEFAULT 0) RETURN t_refcur IS
        l_rc t_refcur;
    BEGIN
        OPEN l_rc FOR
            WITH base AS (
                SELECT e.emp_id,
                    e.emp_name,
                    e.salary,
                    e.bonus_pct,
                    d.dept_name,
                    NVL (
                        (
                        SELECT SUM (s.amount)
                        FROM qt_sales s
                        WHERE s.emp_id = e.emp_id
                    ),
                        0
                    ) AS total_sales
                FROM qt_employees e
                JOIN qt_departments d
                    ON d.dept_id = e.dept_id
                WHERE e.dept_id = p_dept_id
            ),
            ranked AS (
                SELECT b.*,
                    DENSE_RANK () OVER (ORDER BY b.salary DESC NULLS LAST) AS salary_rank
                FROM base b
            )
            SELECT r.emp_id,
                r.emp_name,
                r.dept_name,
                r.salary,
                r.bonus_pct,
                r.total_sales,
                r.salary_rank
            FROM ranked r
            WHERE r.salary >= p_min_salary
            ORDER BY r.salary_rank,
                r.emp_id;
        RETURN l_rc;
    END get_employee_report;

    FUNCTION pipe_sales (p_emp_id NUMBER DEFAULT NULL) RETURN qt_sale_obj_tab PIPELINED IS
    BEGIN
        FOR rec IN (
            SELECT
                s.sale_id,
                s.emp_id,
                s.amount,
                CASE
                    WHEN s.amount < 100 THEN 'LOW'
                    WHEN s.amount BETWEEN 100 AND 500 THEN 'MID'
                    ELSE 'HIGH'
                END AS tag
            FROM qt_sales s
            WHERE p_emp_id IS NULL
                OR s.emp_id = p_emp_id
            ORDER BY s.sale_id
        ) LOOP
            PIPE ROW (qt_sale_obj (rec.sale_id, rec.emp_id, rec.amount, rec.tag));
        END LOOP;
        RETURN;
    END pipe_sales;

    PROCEDURE complex_block (p_dept_id NUMBER, p_raise_pct NUMBER DEFAULT 1.5, p_mode VARCHAR2 DEFAULT 'DEEP') IS
        l_ids t_num_nt;
        l_sql CLOB;
        l_count NUMBER := 0;
        l_rows NUMBER := 0;
        l_mode VARCHAR2 (30) := NVL (p_mode, 'DEEP');
        e_bulk EXCEPTION;
        PRAGMA EXCEPTION_INIT (e_bulk, - 24381);
        CURSOR c_emp (cp_dept_id NUMBER) IS
        SELECT
            emp_id,
            emp_name,
            salary
        FROM qt_employees
        WHERE dept_id = cp_dept_id
        ORDER BY emp_id
        FOR
        UPDATE OF salary SKIP LOCKED;
    BEGIN
        g_run_id := g_run_id + 1;
        AUDIT ('qt_torture_pkg', 'complex_block.begin', 'run=' || g_run_id || '; dept=' || p_dept_id || '; mode=' || l_mode);
        SELECT emp_id BULK COLLECT
        INTO l_ids
        FROM qt_employees
        WHERE dept_id = p_dept_id
        ORDER BY emp_id;
        IF l_ids.COUNT > 0 THEN
            FORALL i IN 1..l_ids.COUNT SAVE EXCEPTIONS
            UPDATE qt_employees
            SET salary = ROUND (NVL (salary, 0) * (1 + p_raise_pct / 100), 2),
                remarks = NVL (remarks, '') ||
                CASE
                    WHEN remarks IS NULL THEN
                    ''
                    ELSE
                    ' | '
                END || 'raised:' || TO_CHAR (SYSTIMESTAMP, 'YYYY-MM-DD HH24:MI:SS.FF3')
            WHERE emp_id = l_ids (i);
        END IF;
        <<outer_loop>>
        FOR r IN c_emp (p_dept_id) LOOP
            DECLARE
                l_local NUMBER := 0;
                l_msg VARCHAR2 (4000);
            BEGIN
                l_local :=
                CASE
                    WHEN r.salary IS NULL THEN
                        - 1
                    WHEN r.salary < 3000 THEN
                        1
                    WHEN r.salary BETWEEN 3000 AND 7000 THEN
                        CASE
                            WHEN MOD (r.emp_id, 2) = 0 THEN
                                2
                            ELSE
                                3
                        END
                    ELSE
                        4
                END;
                l_msg := q'[
BEGIN; this is not a real block delimiter for the outer parser.
It contains semicolons ;;;;; and slashes / / / and comment markers -- /* */
END;
]';
                IF l_mode = 'DEEP' THEN
                    FOR i IN 1..2 LOOP
                        FOR j IN REVERSE 1..2 LOOP
                            EXIT outer_loop WHEN i = 2
                                AND j = 1
                                AND r.emp_id > 1000000;
                            l_count := l_count + i + j + l_local;
                        END LOOP;
                    END LOOP;
                ELSE
                    l_count := l_count + l_local;
                END IF;
                AUDIT ('qt_torture_pkg', 'complex_block.row', 'emp=' || r.emp_id || '; class=' || l_local || '; msg=' || SUBSTR (l_msg, 1, 80));
            EXCEPTION
                WHEN ZERO_DIVIDE THEN
                    AUDIT ('qt_torture_pkg', 'complex_block.zero_divide', 'emp=' || r.emp_id);
                WHEN OTHERS THEN
                    AUDIT ('qt_torture_pkg', 'complex_block.row_error', 'emp=' || r.emp_id || '; err=' || SQLERRM);
            END;
        END LOOP outer_loop;
        l_sql := q'[
            UPDATE qt_sales
               SET note =
                       NVL(note, '') ||
                       CASE
                           WHEN note IS NULL THEN ''
                           ELSE ' | '
                       END ||
                       'mode=' || :b_mode ||
                       '; ts=' || TO_CHAR(SYSTIMESTAMP, 'YYYY-MM-DD"T"HH24:MI:SS')
             WHERE emp_id IN
                   (
                       SELECT emp_id
                       FROM qt_employees
                       WHERE dept_id = :b_dept_id
                   )
        ]';
        EXECUTE IMMEDIATE l_sql USING l_mode,
            p_dept_id;
        l_rows := SQL%ROWCOUNT;
        $IF DBMS_DB_VERSION.VERSION >= 12 $THEN AUDIT ('qt_torture_pkg', 'complex_block.ccflag', 'conditional-compilation=true');
        $ELSE AUDIT ('qt_torture_pkg', 'complex_block.ccflag', 'conditional-compilation=false');
        $END AUDIT ('qt_torture_pkg', 'complex_block.end', 'updated=' || l_rows || '; aggregate=' || l_count);
    EXCEPTION
        WHEN e_bulk THEN
            AUDIT ('qt_torture_pkg', 'complex_block.bulk_error', 'bulk exception raised');
            RAISE;
        WHEN OTHERS THEN
            AUDIT ('qt_torture_pkg', 'complex_block.fatal', SQLERRM);
            RAISE;
    END complex_block;
END qt_torture_pkg;

/* ---------------------------------------------------------------------
   06. COMPOUND TRIGGER
   --------------------------------------------------------------------- */

CREATE OR REPLACE TRIGGER qt_sales_ctrg
    FOR INSERT OR UPDATE OF amount,
    status ON qt_sales COMPOUND TRIGGER g_rows PLS_INTEGER := 0;

    BEFORE STATEMENT IS
    BEGIN
        qt_torture_pkg.AUDIT ('qt_sales_ctrg', 'before statement', 'start');
    END BEFORE STATEMENT;

    BEFORE EACH ROW IS
    BEGIN
        :NEW.status := NVL (:NEW.status, 'NEW');
        IF :NEW.amount < 0 THEN
            RAISE_APPLICATION_ERROR (- 20001, 'Amount must be non-negative');
        END IF;
    END BEFORE EACH ROW;

    AFTER EACH ROW IS
    BEGIN
        g_rows := g_rows + 1;
    END AFTER EACH ROW;

    AFTER STATEMENT IS
    BEGIN
        qt_torture_pkg.AUDIT ('qt_sales_ctrg', 'after statement', 'rows=' || g_rows);
    END AFTER STATEMENT;

END qt_sales_ctrg;
/

CREATE OR REPLACE VIEW qt_complex_v AS
SELECT d.dept_id,
    d.dept_name,
    e.emp_id,
    e.emp_name,
    SUM (s.amount) OVER (PARTITION BY d.dept_id) AS dept_total,
    AVG (s.amount) OVER (PARTITION BY d.dept_id, e.emp_id) AS emp_avg,
    ROW_NUMBER () OVER (PARTITION BY d.dept_id ORDER BY NVL (s.amount, 0) DESC, e.emp_id) AS rn
FROM qt_departments d
JOIN qt_employees e
    ON e.dept_id = d.dept_id
LEFT JOIN qt_sales s
    ON s.emp_id = e.emp_id;
/

/* 09-01. WITH FUNCTION: SQL 안에 로컬 함수 */

WITH FUNCTION score_fn (p_salary NUMBER, p_bonus NUMBER) RETURN NUMBER IS
    BEGIN
        RETURN ROUND (NVL (p_salary, 0) * (1 + NVL (p_bonus, 0) / 100), 2);
    END;

SELECT
    e.emp_id,
    e.emp_name,
    score_fn (e.salary, e.bonus_pct) AS score
FROM qt_employees e
ORDER BY e.emp_id;
/

/* 09-02. Recursive subquery factoring + SEARCH + CYCLE */

WITH dept_tree (dept_id, parent_dept_id, dept_name, lvl, path_txt) AS (
    SELECT
        d.dept_id,
        d.parent_dept_id,
        d.dept_name,
        1 AS lvl,
        '/' || d.dept_name AS path_txt
    FROM qt_departments d
    WHERE d.parent_dept_id IS NULL
    UNION ALL
    SELECT
        c.dept_id,
        c.parent_dept_id,
        c.dept_name,
        p.lvl + 1,
        p.path_txt || '/' || c.dept_name
    FROM qt_departments c
    JOIN dept_tree p
        ON p.dept_id = c.parent_dept_id
) SEARCH DEPTH FIRST BY dept_id
SET dfs_ord CYCLE dept_id
SET is_cycle TO 'Y' DEFAULT 'N'
SELECT
    dept_id,
    parent_dept_id,
    dept_name,
    lvl,
    path_txt,
    dfs_ord,
    is_cycle
FROM dept_tree
ORDER BY dfs_ord;
/

/* 09-03. CONNECT BY NOCYCLE */

SELECT LPAD (' ', 2 * (LEVEL - 1)) || dept_name AS dept_tree,
    dept_id,
    parent_dept_id,
    SYS_CONNECT_BY_PATH (dept_name, '/') AS full_path
FROM qt_departments
START WITH parent_dept_id IS NULL
CONNECT BY NOCYCLE PRIOR dept_id = parent_dept_id
ORDER SIBLINGS BY dept_name;
/

/* 09-04. 엄청난 중첩 쿼리 */

SELECT *
FROM (
        SELECT /*+ qb_name(main_q) leading(e d) use_hash(d) */
            e.emp_id,
            e.emp_name,
            d.dept_name,
            x.total_amount,
            x.avg_amount,
            x.sale_cnt,
            (
                SELECT MAX (s2.amount)
                FROM qt_sales s2
                WHERE s2.emp_id = e.emp_id
            ) AS max_amount,
            CASE
                WHEN x.sale_cnt = 0 THEN 'NO_SALES'
                WHEN x.total_amount > (
                    SELECT AVG (dept_sum)
                    FROM (
                            SELECT SUM (s3.amount) AS dept_sum
                            FROM qt_sales s3
                            JOIN qt_employees e3
                                ON e3.emp_id = s3.emp_id
                            GROUP BY e3.dept_id
                        )
                ) THEN
                CASE
                    WHEN EXISTS (
                        SELECT 1
                        FROM qt_sales sx
                        WHERE sx.emp_id = e.emp_id
                            AND sx.status = 'PAID'
                            AND sx.amount > (
                            SELECT NVL (AVG (sy.amount), 0)
                            FROM qt_sales sy
                            WHERE sy.emp_id = sx.emp_id
                        )
                    ) THEN 'ABOVE_DEPT_AND_SELF_AVG'
                    ELSE 'ABOVE_DEPT_AVG'
                END
                ELSE
                CASE
                    WHEN NVL (x.avg_amount, 0) > 0 THEN
                    CASE
                        WHEN x.avg_amount >= 500 THEN 'MID_PLUS'
                        ELSE 'NORMAL'
                    END
                    ELSE 'NORMAL'
                END
            END AS rank_bucket,
            ROW_NUMBER () OVER (PARTITION BY d.dept_id ORDER BY x.total_amount DESC NULLS LAST, e.emp_id) AS rn
        FROM qt_employees e
        JOIN qt_departments d
            ON d.dept_id = e.dept_id CROSS APPLY (
                SELECT NVL (SUM (s.amount), 0) AS total_amount,
                    NVL (AVG (s.amount), 0) AS avg_amount,
                    COUNT (*) AS sale_cnt
                FROM qt_sales s
                WHERE s.emp_id = e.emp_id
                    AND s.status IN ('NEW', 'PAID', 'PENDING')
            ) x
        WHERE EXISTS (
                SELECT 1
                FROM qt_sales ex
                WHERE ex.emp_id = e.emp_id
                    OR (ex.emp_id = e.emp_id
                        AND ex.note LIKE '%;%')
            )
    ) q
WHERE q.rn <= 3
ORDER BY q.dept_name,
    q.rn,
    q.emp_id;
/

DECLARE
    l_rc SYS_REFCURSOR;
    l_emp_id NUMBER;
    l_emp_name VARCHAR2 (200);
    l_total NUMBER;
    l_sql VARCHAR2 (32767);
BEGIN
    l_sql := q'[
        SELECT
            e.emp_id,
            e.emp_name,
            NVL(SUM(s.amount), 0) AS total_amount
        FROM qt_employees e
        LEFT JOIN qt_sales s
          ON s.emp_id = e.emp_id
         AND s.status <> 'CANCELLED'
        WHERE e.emp_name LIKE :b_name
        GROUP BY e.emp_id, e.emp_name
        HAVING NVL(SUM(s.amount), 0) >= :b_min_total
        ORDER BY e.emp_id
    ]';

        l_sql USING 'A%',
            0;
    LOOP
        FETCH l_rc

            l_emp_name,
            l_total;
        EXIT WHEN l_rc%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE (l_emp_id || ':' || l_emp_name || ':' || l_total);
    END LOOP;
    CLOSE l_rc;
END;
